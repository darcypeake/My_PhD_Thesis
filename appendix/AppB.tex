\section{Initial State Radiation}
In order to extend ARES to initial-state radiation, we follow the same principles as for final-state radiation, namely:
\begin{itemize}
\item we add one collinear emission $k$, and the corresponding virtual
  corrections, on top of hard Born partons, accompanied by an arbitrary
  number of soft and collinear emissions $\{k_i\}$;
\item we subtract the double counting occurring with the soft and collinear region.
\end{itemize}
We use dimensional regularisation to make each contribution
individually finite. In the case of initial-state radiation, virtual
corrections include the counterterm arising from the renormalisation
of PDFs. \\ \\
To be concrete, we consider the case of an incoming quark carrying a fraction $x/z$ of the proton momentum. This emits a
collinear gluon carrying a fraction $1-z$ of its emitter's momentum. After the emission, the quark enters the hard scattering with a
fraction $x$ of the proton momentum. The above procedure gives a
function that we call $\mathcal{F}_{\mathrm{ISR}}$. This function will be
reorganised to give various NNLL functions. Our starting point is
\begin{equation}
  \label{eq:FISR-start}
  \begin{split}
    \mathcal{F}_{\mathrm{ISR}} & = \int \dZ \left\{\int_0^{Q^2}\frac{dk_t^2}{k_t^2}\left(\frac{\mu_R^2}{k_t^2}\right)^{\varepsilon} \frac{\alpha_s(k_t)}{2\pi}\times\right. \\& \times \left[\int_x^1 \frac{dz}{z}C_F \left[\frac{1+z^2}{1-z}-\varepsilon(1-z)\right]\Theta\left(1-\frac{V_{\mathrm{hc}}(\{\tilde{p}\},k,\{k_i\}}{v}\right) \frac{f_q(x/z,\mu_F)}{f_q(x,\mu_F)} + \right. \\
    & \left.- \int_0^1 dz \frac{2C_F}{1-z}\Theta\left(1-\frac{V_{\mathrm{sc}}(\{\tilde{p}\},k,\{k_i\}}{v}\right)\right] + \\
   & + \left[-\gamma_q^{(0)}\int_0^{Q^2v^{2/(a+b_q)}}\frac{dk^2}{k^2}\left(\frac{\mu_R^2}{k^2}\right)^{\varepsilon}\frac{\alpha_s(k)}{2\pi}
   +\frac{1}{\varepsilon} \left(\frac{\mu_R^2}{\mu_F^2}\right)^{\varepsilon} \frac{\alpha_s(\mu_R)}{2\pi} \int_x^1 \frac{dz}{z} C_F\left(\frac{1+z^2}{1-z}\right)_+ \frac{f_q(x/z,\mu_F)}{f_q(x,\mu_F)}\right] \times \\ & \times \left. \Theta\left(1-\frac{V_{\mathrm{sc}}(\{\tilde{p}\},\{k_i\}}{v}\right)\right\} \,.
\end{split}
\end{equation}

The expression in eq.~\eqref{eq:FISR-start} will contain single logarithms of the form
$\ln[Qv^{1/(a+b_q)}/\mu_R]$ and $\ln[Qv^{1/(a+b_q)}/\mu_F]$. This
logarithms can be resummed by fixing $\mu_R=\mu_F=Qv^{1/(a+b_q)}$. Also, we can set $\alpha_s(k)=\alpha_s(k_t)=\alpha_s(Qv^{1/(a+b_q)})$ and obtain a finite expression, up to subleading terms of order $\alpha_s^2$. \\ \\
In order for every term within $\mathcal{F}_{\mathrm{ISR}}$ to be finite in four dimensions, we must add and subtract a counterterm, $\mathrm{CT}$, which has the following form:
\begin{equation}
  \begin{aligned}
    \mathrm{CT}&=\int \dZ \int\frac{dk_t^2}{k_t^2}\left(\frac{\mu_R^2}{k_t^2}\right)^{\varepsilon} \frac{\alpha_s(k_t)}{2\pi}\times \\
    &\times C_F\left[\int_x^1\frac{dz}{z}\left(\frac{1+z^2}{1-z}-\varepsilon(1-z)\right)\frac{f_q(x/z)}{f_q(x)}-\int_0^1dz\frac{2}{1-z}\right]\times\\
    &\times\Theta\left(1-\frac{V_{\mathrm{sc}}(\{\tilde{p}\},k)}{v}\right)\Theta\left(1-\frac{V_{\mathrm{sc}}(\{\tilde{p}\},\{k_i\}}{v}\right)\,.
  \end{aligned}
\end{equation}

\begin{equation}
  \label{eq:FISR-split}
  \mathcal{F}_{\mathrm{ISR}} = \frac{\alpha_s}{\pi}\left( \delta \mathcal{F}_{\mathrm{rec,q}}+\delta\mathcal{F}_{\mathrm{hc,q}}\right)+ \frac{\alpha_s(Q v^{1/(a+b_q)})}{2\pi} C^{(1)}_{\mathrm{hc,q}} \mathcal{F}_{\mathrm{NLL}}\,.

\noindent We are now in a position to calculate the initial-state radiation contributions to the $q\bar{q}$ channel: $C^{(1)}_{\mathrm{hc,q}}$, $\delta \mathcal{F}_{\mathrm{rec,q}}$, and $\delta \mathcal{F}_{\mathrm{hc,q}}$. The hard-collinear coefficient is
\begin{equation}
\begin{aligned}
C^{(1)}_{\mathrm{hc}, \ell}&= \int_x^1\frac{dz}{z}C_F
\left[ \left(\frac{2 }{a+b_\ell}\left(b_\ell\left(\frac{\ln(1-z)}{1-z}\right)_+-\frac{\ln d_\ell g_\ell(\phi)}{(1-z)_+}\right)\right)(1+z^2)+(1-z)\right]\frac{f_q(x/z,Qv^{\frac{1}{a+b_q}})}{f_q(x,Qv^{\frac{1}{a+b_q}})}+\\
&+\int_x^1\frac{dz}{z}T_R
\left[\left(\frac{2}{a+b_\ell}\left(b_\ell\ln(1-z)
-\ln d_\ell g_\ell(\phi)\right)\left((1-z)^2+z^2\right)\right)
+2z(1-z)\right]\frac{f_g(x/z,Qv^{\frac{1}{a+b_q}})}{f_q(x,Qv^{\frac{1}{a+b_q}})}\,.
\end{aligned}
\end{equation}
The remaining terms can be expressed more generally. The recoil correction is
\begin{equation}
\begin{aligned}
\delta \mathcal{F}_{\mathrm{rec,q}}& =\int \dZ \frac{\alpha_s\left(Q v^{1/(a+b_\ell)}\right)}{\alpha_s(Q)\left(a+b_\ell\right)}\left\{\int_0^\infty\frac{d\zeta}{\zeta}\int_0^\pi\frac{d\phi}{\pi}\int_0^1 dz\times\right. \\
&\times P^{\mathrm{rec}}_q(z)
\left[\Theta\left(1-\frac{V_{\mathrm{hc}}(\{\tilde{p}\},k,\{k_i\})}{v}\right)-\Theta\left(1-\frac{V_{\mathrm{sc}}(\{\tilde{p}\},k,\{k_i\})}{v}\right)\right]\,,
\end{aligned}
\end{equation}
where
\begin{equation}
  P^{\mathrm{rec}}_q(z) = \frac{1}{z}\frac{1}{f_q(x,\mu_F)}\left[ f_q(x/z,\mu_F)C_F\frac{1+z^2}{1-z}+ f_g(x/z,\mu_F)T_R(z^2+(1-z)^2)\right]\Theta\left(z-x\right)\,.
\end{equation}
For final-state radiation, this reduces to
\begin{equation} 
\label{eq:P-avg}
 P^{\mathrm{rec}}_{f_\ell}(z)\equiv
   \begin{cases}
    C_F\frac{1+z^2}{1-z}\,, & f_\ell = q\,,\\
    C_A\left(\frac{2}{1-z}+z(1-z)-2\right)+T_R n_f\left(1-2z(1-z)\right)\,, & f_\ell = g\,.
  \end{cases}
\end{equation}
Finally, the hard-collinear correction is
\begin{equation}
\begin{aligned}
\delta \mathcal{F}_{\mathrm{hc,q}}& = \int \dZ\frac{\alpha_s\left(v^{1/(a+b_\ell)}\right)}{\alpha_s(Q)\left(a+b_\ell\right)}\int_0^\infty\frac{d\zeta}{\zeta}\int_0^\pi\frac{d\phi}{\pi}\int_0^1 dz  \, P^{\mathrm{hc}}_q(z) \\
&\times \left[\Theta\left(1-\frac{V_{\mathrm{sc}}(\{\tilde{p}\},k,\{k_i\})}{v}\right) -\Theta(1-\zeta)\Theta\left(1-\frac{V_{\mathrm{sc}}(\{\tilde{p}\},\{k_i\})}{v}\right)\right]\,,
\end{aligned}
\end{equation}
with
\begin{equation}
P^{\mathrm{hc}}_q(z)=\frac{1}{z}\frac{1}{f_q(x,\mu_F)}\left[ f_q(x/z,\mu_F)C_F\frac{1+z^2}{(1-z)_{+}}+ f_g(x/z,\mu_F)T_R(z^2+(1-z)^2)\right]\Theta\left(z-x\right)\,.
\end{equation}
In the context of final-state radiation, $P^{\mathrm{hc}}_q(z)\equiv P^{\mathrm{rec}}_q(z)$. \\ \\

\subsubsection{Averaged Contributions}
Analogously to sec.~\ref{subsec:ISR-qqZg}, we will now discuss each of the contributions making up the ISR component of the $gq\to Z+q$ channel. The hard-collinear coefficient is:
\begin{equation}
\begin{aligned}
C^{(1)}_{\mathrm{hc}, \ell}&= \int_x^1\frac{dz}{z}2C_A
\left[ \left(\frac{2 }{a+b_\ell}\left(b_\ell\left(\frac{\ln(1-z)}{1-z}\right)_+-\frac{\ln d_\ell g_\ell(\phi)}{(1-z)_+}\right)\right)\frac{(1-z+z^2)^2}{z}\right]\frac{f_g(x/z,Qv^{\frac{1}{a+b_q}})}{f_g(x,Qv^{\frac{1}{a+b_q}})}+\\
&+\int_x^1\frac{dz}{z}C_F
\left[\left(\frac{2}{a+b_\ell}\left(b_\ell\ln(1-z)
-\ln d_\ell g_\ell(\phi)\right)\frac{1+(1-z)^2}{z}\right)
+z\right]\frac{f_g(x/z,Qv^{\frac{1}{a+b_q}})}{f_q(x,Qv^{\frac{1}{a+b_q}})}\,.
\end{aligned}
\end{equation}
Again, we find the recoil correction to be
\begin{equation}
\begin{aligned}
\delta \mathcal{F}_{\mathrm{rec,g}}& =\int \dZ \frac{\alpha_s\left(Q v^{1/(a+b_\ell)}\right)}{\alpha_s(Q)\left(a+b_\ell\right)}\left\{\int_0^\infty\frac{d\zeta}{\zeta}\int_0^\pi\frac{d\phi}{\pi}\int_0^1 dz\times\right. \\
&\times P^{\mathrm{rec}}_q(z)
\left[\Theta\left(1-\frac{V_{\mathrm{hc}}(\{\tilde{p}\},k,\{k_i\})}{v}\right)-\Theta\left(1-\frac{V_{\mathrm{sc}}(\{\tilde{p}\},k,\{k_i\})}{v}\right)\right]\,,
\end{aligned}
\end{equation}
where
\begin{equation}
  P^{\mathrm{rec}}_g(z) = \frac{1}{z}\frac{1}{f_g(x,\mu_F)}\left[ f_g(x/z,\mu_F)2C_A\frac{(1-z+z^2)^2}{z(1-z)}+ f_q(x/z,\mu_F)C_F\frac{(1+(1-z)^2)}{z}\right]\Theta\left(z-x\right)\,.
\end{equation}
Finally, the hard-collinear correction is
\begin{equation}
\begin{aligned}
\delta \mathcal{F}_{\mathrm{hc,g}}& = \int \dZ\frac{\alpha_s\left(v^{1/(a+b_\ell)}\right)}{\alpha_s(Q)\left(a+b_\ell\right)}\int_0^\infty\frac{d\zeta}{\zeta}\int_0^\pi\frac{d\phi}{\pi}\int_0^1 dz  \, P^{\mathrm{hc}}_q(z) \\
&\times \left[\Theta\left(1-\frac{V_{\mathrm{sc}}(\{\tilde{p}\},k,\{k_i\})}{v}\right) -\Theta(1-\zeta)\Theta\left(1-\frac{V_{\mathrm{sc}}(\{\tilde{p}\},\{k_i\})}{v}\right)\right]\,,
\end{aligned}
\end{equation}
with
\begin{equation}
P^{\mathrm{hc}}_g(z)= \frac{1}{z}\frac{1}{f_g(x,\mu_F)}\left[ f_g(x/z,\mu_F)2C_A\frac{(1-z+z^2)^2}{z(1-z)_{\!+}}+ f_q(x/z,\mu_F)C_F\frac{(1+(1-z)^2)}{z}\right]\Theta\left(z-x\right)\,.
\end{equation}

\end{equation}

\begin{aligned}

\subsubsection{Un-averaged Contributions}
The spin-density matrix is:
\begin{equation}
	\rho^{\mu\nu}=-\left(\frac{1}{2}g_T^{\mu\nu}+\mathcal{P}_L\left(\frac{p_T^\mu p_T^\nu}{p_T^2}-\frac{1}{2}g_T^{\mu\nu}\right)+\mathcal{P}_C^{\mu\nu}\epsilon_T^{\mu\nu}\right)\,,
\end{equation}
where
\begin{equation}
g_T^{\mu\nu}=g^{\mu\nu}-\frac{p_1^\mu p_2^\nu+p_2^\mu p_1^\nu}{p_1\cdot p_2}\,,
\end{equation}
and $\mathcal{P}_{L}$ and $\mathcal{P}_{C}$ are defined as the following for the $gq\to Zq$ channel
\begin{equation}
\begin{aligned}
	&\mathcal{P}_L=\frac{2su}{s^2+u^2+2M^2t}\,,\\
	&\mathcal{P}_C=\frac{g_Vg_A}{g_V^2+g_A^2}\frac{s^2-u^2}{s^2+u^2+2M^2t}\,.
\end{aligned}
\end{equation}
In order to find $\Delta P_g$, we must contract the tensor $\rho^{\mu\nu}$ with the following splitting functions
\begin{equation}
\begin{aligned}
 	&\left[\hat{P}_{gg}^{(1)}(z,\mathbf{p_T})\right]^{\mu\nu}=2C_A\left[-g^{\mu\nu}\left(\frac{z}{1-z}+z(1-z)\right)+2\frac{p_T^\mu p_T^\nu}{\mathbf{q_T}^2}\frac{1-z}{z}\right]\,,\\
	&\left[\hat{P}_{gq}^{(1)}(z,\mathbf{p_T})\right]^{\mu\nu}=\left[\hat{P}_{g\bar{q}}^{(1)}(z,\mathbf{p_T})\right]^{\mu\nu}=2C_F\left[-g^{\mu\nu}\frac{z}{2}+2\frac{p_T^\mu p_T^\nu}{\mathbf{q_T}^2}\frac{1-z}{z}\right]\,.
\end{aligned}	
\end{equation}
In fact, we find that it is convenient to write these gluon splitting functions as:
\begin{equation}
	\left[\hat{P}_{ga}^{(1)}(z,\mathbf{p_T})\right]^{\mu\nu}=-g^{\mu\nu}\hat{P}_{ga}^{(1)}+\left(g^{\mu\nu}+2\frac{q_T^\mu q_T^\nu}{\mathbf{q_T}^2}2G_{ga}^{(1)}(z)\right)\,,\quad a=g,q,\bar{q}\,,
\end{equation}
where
\begin{equation}
\begin{aligned}
	&G_{gg}^{(1)}=C_A\frac{1-z}{z}\,,\quad\\
	&G_{gq}^{(1)}=G_{g\bar{q}}^{(1)}=C_F\frac{1-z}{z}\,,
\end{aligned}
\end{equation}
and $\hat{P}_{ga}^{(1)}$ are the azimuthally averaged splitting functions:
\begin{equation}
\begin{aligned}
	&\hat{P}_{gg}^{(1)}=2C_A\left[\frac{z}{1-z}+\frac{1-z}{z}+z(1-z)\right]\,,\\
	&\hat{P}_{gq}^{(1)}=\hat{P}_{g\bar{q}}^{(1)}=2C_F\frac{1+(1-z)^2}{z}\,.
\end{aligned}
\end{equation}
From here, we find that the contraction is:
\begin{equation}
	\rho^{\mu\nu}\left[\hat{P}_{ga}^{(1)}(z,\mathbf{p_T})\right]_{\mu\nu}=\hat{P}_{ga}^{(1)}+2\mathcal{P}_LG_{ga}^{(1)}\left[2\cos^2\phi-1\right]\,,\quad a=g,q,\bar{q}\,.
\end{equation}
The $\hat{P}_{ga}^{(1)}$ term contributes to the $\delta\mathcal{F}_{\mathrm{rec}}$ while the spin correlation effects come from the term dependent on $G_{ga}^{(1)}$ such that we find:
\begin{equation}
\begin{aligned}
\Delta\mathcal{F}_{\mathrm{rec}}& =\frac{\alpha_s\left(Q v^{1/(a+b_\ell)}\right)}{\alpha_s(Q)\left(a+b_\ell\right)}\int_0^\infty\frac{d\zeta}{\zeta}\int_0^\pi\frac{d\phi}{\pi}\int_x^1 \frac{dz}{z}\frac{1}{f_g(x,\mu_F)}\Delta P_{g}(z)\times\\&\times\int \dZ \left[\Theta\left(1-\frac{V_{\mathrm{hc}}(\{\tilde{p}\},k,\{k_i\})}{v}\right)-\Theta(1-\zeta)\Theta\left(1-\frac{V_{\mathrm{sc}}(\{\tilde{p}\},\{k_i\})}{v}\right)\right]\,,
\end{aligned}
\end{equation}
where
\begin{equation}
\begin{aligned}
\Delta P_g =2\mathcal{P}_L\left(\frac{1-z}{z}\right)\left[2\cos^2\phi-1\right](C_Af_g(x/z,\mu_F)+C_Ff_q(x/z,\mu_F))\,.
\end{aligned}
\end{equation}
We will see how this is calculated for the observables:
\paragraph{One-jettiness}
There is no $\phi$-dependence for this observable and so $2\cos^2\phi-1$ integrates to zero such that:
\begin{equation}
	\Delta\mathcal{F}_{\mathrm{rec}}=0\,.
\end{equation}
\paragraph{Thrust minor}
For this observable, we find that it is half the value for the D-parameter as shown in Ref.~\cite{Arpino:2019ozn} due to for Thrust minor $g_\ell(\phi)=|\sin\phi|$, while for the D-parameter it is $g_\ell(\phi)=\sin^2\phi$. This yields result of
\begin{equation}
	\Delta\mathcal{F}_{\mathrm{rec}}=\FNLL(\lambda)\frac{\alpha_s\left(\sqrt{\tau_m}Q\right)}{2\alpha_s(Q)}\int_x^1\frac{dz}{z}\frac{1}{f_g(x,\mu_F)} \mathcal{P}_L\left(\frac{1-z}{z}\right)(C_Af_g(x/z,\mu_F)+C_Ff_q(x/z,\mu_F))
\end{equation}
