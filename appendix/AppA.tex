The generalised-$k_t$ jet algorithm for DIS presented in
Chapter~\ref{chapter:DIS} has been implemented as the
\href{https://repo.hepforge.org/source/fastjetsvn/browse/contrib/contribs/DISGenkt}{\texttt{DISGenkt}}
plugin to \textsc{FastJet} as part of \texttt{fjcontrib}.
Within the implementation, the pairwise distance $d_{ij}$ and
beam distance $d_{iB}$ given in eq.~\eqref{eq:dij-diB-ouralg}
are evaluated by the \texttt{DISBriefJet} class. The clustering
loop uses the \textsc{FastJet} \texttt{NNH} (nearest-neighbour
heuristic) class, which internally calls \texttt{distance()} and
\texttt{beam\_distance()} to find the closest pair of objects at
each step, returning \texttt{j=-1} when the minimum is a beam
distance $d_{iB}$. The identification of the macrojet is
determined by finding the jet with the largest DIS longitudinal
projection $E-\mathrm{beam\_sign}\cdot p_z$, with an additional
function to sort jets by decreasing DIS longitudinal projection.
These are implemented as follows.

\begin{lstlisting}[caption={Pairwise distance $d_{ij} =
\min(E_i^{2p},E_j^{2p})(1-\cos\theta_{ij})/(1-\cos R)$ between
two jets.}]
double DISBriefJet::distance(const DISBriefJet * jet) const {
  double dij = 1 - _nx*jet->_nx
                 - _ny*jet->_ny
                 - _nz*jet->_nz;
  dij = dij * fmin(_E2p, jet->_E2p) / _dij_norm;
  return dij;
}
\end{lstlisting}



\begin{lstlisting}[caption={Beam distance $d_{iB} =
E_i^{2p}(1-\cos\theta_{iB})$ between a jet and the proton beam.}]
double DISBriefJet::beam_distance() const {
  return (1 - _beam_sign * _nz)*_E2p;
}
\end{lstlisting}

\begin{lstlisting}[caption={Main clustering loop. At each step the
minimum distance is found via the \texttt{NNH} nearest-neighbour
heuristic. If the minimum is $d_{ij}$ ($j\geq0$), jets $i$ and
$j$ are merged, whereas if it is $d_{iB}$ ($j<0$), jet $i$ is declared
final and removed from the clustering.}]
void DISGenktPlugin::run_clustering(
    fastjet::ClusterSequence & cs) const {
  int njets = cs.jets().size();
  DISjetInfo info(p(), beam_sign(), R());
  NNH<DISBriefJet,DISjetInfo> nnh(cs.jets(), &info);

  while (njets > 0) {
    int i, j, k;
    double dij = nnh.dij_min(i, j);
    if (j >= 0) {
      cs.plugin_record_ij_recombination(i, j, dij, k);
      nnh.merge_jets(i, j, cs.jets()[k], k);
    } else {
      cs.plugin_record_iB_recombination(i, dij);
      nnh.remove_jet(i);
    }
    njets--;
  }
}
\end{lstlisting}


\begin{lstlisting}[caption={Find the jet with the largest DIS
longitudinal projection $E_i - \mathrm{beam\_sign}\cdot p_{iz}$.
This jet is identified as the macrojet.}]
int DISGenktPlugin::find_idx_macrojet(
    const std::vector<PseudoJet>& jets) const {
  double best = 0.0;
  int idx = -1;
  for (size_t i = 0; i < jets.size(); ++i) {
    double proj = jets[i].E() - beam_sign() * jets[i].pz();
    if (proj > best) {
      best = proj;
      idx  = static_cast<int>(i);
    }
  }
  return idx;
}
\end{lstlisting}

\begin{lstlisting}[caption={Sort jets by decreasing DIS longitudinal
projection, returning a set of jets from largest to smallest
$E - \mathrm{beam\_sign}\cdot p_{z}$.}]
std::vector<PseudoJet> DISGenktPlugin::sorted_by_zjet(
    const std::vector<PseudoJet> & jets) const {
  std::vector<double> zproj(jets.size());
  for (size_t i = 0; i < jets.size(); i++) {
    zproj[i] = -(jets[i].E() - beam_sign() * jets[i].pz());
  }
  return objects_sorted_by_values(jets, zproj);
}
\end{lstlisting}
