\section{Spin correlations for ISR}
\label{app:spin-correl}
In order to compute hard-collinear NNLL corrections, we need to take into account spin correlations. The starting point is the spin-density matrix for the hard process, in $4-2\varepsilon$ dimensions. Neglecting terms that are antisymmetric in the indexes $\mu,\nu$, this has the general form
\begin{equation}
\rho^{\mu\nu} = \frac{d^{\mu\nu} }{2(1-\varepsilon)}
+ \mathcal{T}_\ell(\{p\};\varepsilon)\left(2n_{\rm in}^\mu n_{\rm in }^\nu - \frac{d^{\mu\nu} }{1-\varepsilon}\right) \,.
\end{equation}
We choose $d^{\mu\nu}$ to be the sum over physical polarisations transverse to both $p_1$ and $p_2$:
\begin{equation}
d^{\mu\nu} = -g^{\mu\nu} + \frac{p_1^\mu p_2^\nu + p_2^\mu p_1^\nu}{p_1 \cdot p_2}\,,
\end{equation}
The space-like vector $n_{\rm in}$ lies in the plane identified by the beam and the $Z$ boson, and satisfies $n_{\rm in}^2=-1$ and $n_{\rm in}\cdot p_1=n_{\rm in}\cdot p_2=0$. We obtain the following relations: 
\begin{equation}
  \begin{split}
  &d^{\mu\nu}d_{\mu\nu}=-g^{\mu\nu}d_{\mu\nu}=2(1-\varepsilon)\,,\\ & d_{\mu\nu} \left(2n_{\rm in}^\mu n_{\rm in }^\nu - \frac{d^{\mu\nu}}{1-\varepsilon}\right)=-g^{\mu\nu}\left(2 n_{\rm in}^\mu n_{\rm in }^\nu - \frac{d^{\mu\nu}}{1-\varepsilon}\right)=0\,,\\ &
  \left(2 n_{\rm in}^\mu n_{\rm in }^\nu - \frac{d^{\mu\nu}}{1-\varepsilon}\right)\left(2 n_{{\rm in},\mu} n_{{\rm in },\nu} - \frac{d^{\mu\nu}}{1-\varepsilon}\right) = 2 \frac{1-2\varepsilon}{1-\varepsilon}\,.
  \end{split}
\end{equation}
This implies
\begin{equation}
  \label{eq:PL}
  \mathcal{T}_\ell(\{\tilde p\};\varepsilon) = \frac{1-\varepsilon}{2(1-2\varepsilon)} \left(2 n_{\rm in}^\mu n_{\rm in }^\nu - \frac{d^{\mu\nu}}{1-\varepsilon}\right) \rho_{\mu\nu}\,.
\end{equation}
