\begin{itemize}
\item \textbf{The NNLL corrections.} The soft-collinear correction is
	\begin{multline}
	\label{eq:dFsc-LTS}
	\delta\mathcal{F}^{\rm LTS}_{\mathrm{sc}}(v)=-\frac{\pi}{\alpha_s(\mu_R)}\FNLL(v)
	\sum_{\ell=1,2}
	\left[\delta R'_{\mathrm{NNLL},\ell}(v)\left(\psi^{(0)}(1+R'_{\mathrm{NLL}}(v))+\gamma_E\right)\right.\\
	\left.+\frac{R''_{\ell}(v)}{2}\left(
	\left(\psi^{(0)}(1+R'_{\mathrm{NLL}}(v))+\gamma_E\right)^2
	-\psi^{(1)}(1+R'_{\mathrm{NLL}}(v))+\frac{\pi^2}{6}\right)\right]\,.
	\end{multline}
	The wide-angle correction vanishes, $\delta\mathcal{F}^{\rm LTS}_{\mathrm{wa}}(v)=0$. 
	the correlated correction is
	\begin{equation}
	\label{eq:dFcorrel-LTS}
	\delta\mathcal{F}_{\mathrm{correl}}(v)=
	\mathcal{F}_{\mathrm{NLL}}
	\frac{\lambda R''(v)}{\alpha_s(\mu_R)\beta_0}
	\left(
	C_A\langle\ln f_{\mathrm{correl}}\rangle_{C_A}
	+n_f\langle\ln f_{\mathrm{correl}}\rangle_{n_f}
	\right)\,,
	\end{equation}
	with $\langle \ln f_{\mathrm{correl}}\rangle_{C_A}$ and $\langle \ln
	f_{\mathrm{correl}}\rangle_{n_f}$ given in
	Ref.~\cite{vanBeekveld:2025zjh}, and the clustering correction is
	\begin{equation}
	\label{eq:dFclust-LTS}
	\delta\mathcal{F}_{\mathrm{clust}}(v)=
	\mathcal{F}_{\mathrm{NLL}}
	\frac{\lambda R''(v)}{\alpha_s(\mu_R)\beta_0}
	C_{\ell}\langle \ln f_{\mathrm{clust}}\rangle\,,
	\end{equation}
	with $\langle \ln f_{\mathrm{clust}}\rangle$ likewise given in
	Ref.~\cite{vanBeekveld:2025zjh}. 
\end{itemize}

Here we present the components used to build the subtraction ingredients
of Ref.~\cite{Gaunt:2015pea}. Legs are labelled $a$, $b$ (incoming) and
$c$ (outgoing jet), corresponding to legs $\ell=1,2,3$ of the main

$Q_a, Q_b, Q_c$ carry the leg label throughout. The dummy flavour
summed over in each beam function is denoted $k$, running over all
parton flavours that can produce $a$ or $b$ via a collinear splitting
--- for example, if $a=q$ then $k\in\{q,g\}$, corresponding to the
$P_{qq}^{(0)}$ and $P_{qg}^{(0)}$ splitting channels respectively.

    -\ln\frac{Q_{ab}^2}{Q_a Q_b}\right)\right]\,.
\end{align}
\end{subequations}
