The one-loop virtual corrections
to $q\bar{q}\to Zg$ are not directly available in the literature
and must be obtained via analytic continuation from $e^+e^-$, which
is technically involved. We therefore use Ref.~\cite{Ravindran:2002dc},
which computes the one-loop corrections to $q\bar{q}\to gH$. This
process shares the same three-leg colour structure as $q\bar{q}\to Zg$
and therefore has identical IR pole structure; only the
process-specific finite constant $\mathcal{H}_1$ differs. We
demonstrate the extraction for the $q\bar{q}\to Zg$ channel; the
remaining channels $qg\to Hq$ and $gq\to Hq$ follow
similarly.


\paragraph{Pole structure from \textsc{ARES}/Catani}
Expanding Eq.~\eqref{eq:HQ-expanded} in $\varepsilon$ for the
$q\bar{q}\to Zg$ process, the double-pole
coefficient is
\begin{equation}
  \label{eq:H-pole2}
  -\frac{1}{\varepsilon^2}(C_A+2C_F)\,,
\end{equation}
and the single-pole coefficient is
\begin{equation}
  \label{eq:H-pole1}
  \frac{1}{\varepsilon}\!\left(
  -\frac{11}{6}C_A-3C_F+\frac{n_f}{3}
  +(C_A-2C_F)\ln\frac{\mu_R^2}{Q_{12}^2}
  -C_A\ln\frac{\mu_R^2}{Q_{13}^2}
  -C_A\ln\frac{\mu_R^2}{Q_{23}^2}
  \right)\,.
\end{equation}

\paragraph{Comparison with Ref.~\cite{Ravindran:2002dc}}
Now comparing with Ref.~\cite{Ravindran:2002dc}. The bare cross section at leading-order is
\begin{equation}
  s^2\frac{d^2\sigma_{q\bar{q}\to g H}^{(1)}}{dt\,du} =
  \pi\delta(s+t+u-M^2)\,
  \frac{S_\varepsilon}{\Gamma(1-\varepsilon)}
  \,\frac{1}{4N_c^2}
  \left(\frac{\mu_0^2\,s}{tu}\right)^\varepsilon
 \left( \frac{\alpha_s(\mu_0^2)}{4\pi}\right)^3|M_{q\bar{q}\to g H}^{(1)}|^2
  \,,
\end{equation}
where $S_\varepsilon=\exp(\varepsilon(\ln4\pi-\gamma_E))$ and $\mu_0$ is the bare reference scale.  

We must renormalise the strong coupling constant due to UV divergences using,
\begin{equation}
  \label{eq:Catani-renorm}
  \alpha_s(\mu_0^2) =
\alpha_s(\mu_R^2)
  \!\left[1-\frac{\alpha_s(\mu_R^2)\beta_0}{\varepsilon}
  +\mathcal{O}(\alpha_s^2)\right] \left(\frac{\mu_R^2}{\mu_0^2}\right)^\varepsilon\,,
\end{equation}
leading to the renormalised form of the LO cross-section
\begin{equation}
\begin{aligned}
  \label{eq:LO-ren}
  s^2\frac{d^2\sigma_{q\bar{q}\to gH}^{{\rm LO},\rm Ren}}{dt\,du}& \simeq
  \pi\delta(s-t-u-M^2)\,
  \frac{S_\varepsilon}{\Gamma(1-\varepsilon)}
  \,\frac{1}{4N_c^2}
  \left(\frac{\mu_R^2\,s}{tu}\right)^\varepsilon
\left(\frac{\alpha_s(\mu_R^2)}{4\pi}\right)^3
\left(\frac{\mu_R^2}{\mu_0^2}\right)^{2\varepsilon}\\
&\times  \!\left[1-\frac{3\alpha_s(\mu_R^2)\beta_0}{\varepsilon}\right] |M_{q\bar{q}\to gH}^{(0)}|^2\,.
  \end{aligned}
  \end{equation}

\paragraph{Pole comparison} We find that the poles within the square brackets of Eq.~\eqref{eq:LO+virt-ren}
are:
\begin{equation}
  \label{eq:vN-pole2}
  -\frac{2}{\varepsilon^2}(C_A+2C_F)\,,
\end{equation}
and the single-pole coefficient is
\begin{equation}
   \label{eq:vN-pole1}
  \frac{1}{\varepsilon}\!\left(
  -\frac{11}{3}C_A-6C_F+\frac{2n_f}{3}
  +2(C_A-2C_F)\ln\frac{\mu_R^2}{Q_{12}^2}
  -2C_A\ln\frac{\mu_R^2}{Q_{13}^2}

  \right)\,.
\end{equation}

Comparing with Eqs.~\eqref{eq:H-pole2} and~\eqref{eq:H-pole1},
the poles of Ref.~\cite{Ravindran:2002dc} are exactly twice those
of $\mathcal{V}(Q;\varepsilon)$, this factor is taken into account when 
subtracting the finite terms.

\paragraph{Extraction of the finite remainder}
The finite remainder is extracted from the
$\mathcal{O}(\varepsilon^0)$ terms of Eq.~\eqref{eq:VIRT-bare}
after subtracting the predicted poles giving rise to
the SCET hard function
\begin{equation}
\label{eq:C1-SCET}
\begin{aligned}
& H^{(1)}_{\rm SCET}=\;\\
& C_A\left[
\frac{76}{9}
+2\operatorname{Li}_2\left(\frac{Q_{12}^2-M^2}{Q_{12}^2}\right)
-\frac{11}{3}\ln\frac{Q_{12}^2}{\mu_R^2}
+\frac{1}{2}\ln^2\frac{\mu_R^2}{Q_{12}^2}
-\frac{1}{2}\ln^2\frac{\mu_R^2}{Q_{13}^2}
-\frac{1}{2}\ln^2\frac{\mu_R^2}{Q_{23}^2}
+\ln\frac{Q_{12}^2}{Q_{13}^2}
 \ln\frac{Q_{12}^2}{Q_{23}^2}
\right]\\
&+C_F\left[-8
+2\operatorname{Li}_2\left(\frac{-Q_{13}^2}{M^2}\right)
+2\operatorname{Li}_2\left(\frac{-Q_{23}^2}{M^2}\right)
-\ln^2\frac{-Q_{13}^2-M^2}{-Q_{13}^2}
-\ln^2\frac{-Q_{23}^2-M^2}{-Q_{23}^2}\right.\\
&\qquad\left.
+\ln^2\frac{M^2+Q_{13}^2}{M^2}
+\ln^2\frac{M^2+Q_{23}^2}{M^2}

+3\ln\frac{Q_{12}^2}{\mu_R^2}
+\ln^2\frac{Q_{13}^2}{Q_{23}^2}
\right]\\
&+n_f\left(-\frac{10}{9}-\frac{2}{3}\ln\frac{\mu_R^2}{Q_{12}^2}\right)+\pi^2\left(
3C_F-\frac{7}{6}C_A
\right)\,.
\end{aligned}
\end{equation}
