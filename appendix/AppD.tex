Here we extract the one-loop hard function $H^{(1)}_{\rm SCET}$ for SCET
and the finite $\mathcal{H}_1$ piece from the hard coefficient $H^{(1)}_{ij}$ in
\textsc{ARES} from the one-loop virtual amplitude of Ref.~\cite{Ravindran:2002dc}. 
These terms enter the coefficients $h_{10}, h_{22},$ and $h_{21}$.

The one-loop virtual corrections
to $q\bar{q}\to Zg$ are not directly available in the literature
and must be obtained via analytic continuation from $e^+e^-$, which
is technically involved. We therefore use Ref.~\cite{Ravindran:2002dc},
which computes the one-loop corrections to $q\bar{q}\to gH$. This
process shares the same three-leg colour structure as $q\bar{q}\to Zg$
and therefore has identical IR pole structure; only the
process-specific finite constant $\mathcal{H}_1$ differs. We
demonstrate the extraction for the $q\bar{q}\to Zg$ channel; the
remaining channels $qg\to Hq$ and $\bar{q}g\to H\bar{q}$ follow
by crossing symmetry.


\paragraph{Pole structure from \textsc{ARES}/Catani}
Expanding eq.~\eqref{eq:HQ-expanded} in $\varepsilon$ for the
$q\bar{q}\to Zg$ process, with dipole colour factors
$C_{12}=C_F-C_A/2$, $C_{13}=C_{23}=C_A/2$, the double-pole
coefficient is
\begin{equation}
  \label{eq:H-pole2}
  -\frac{1}{\varepsilon^2}(C_A+2C_F)\,,
\end{equation}
and the single-pole coefficient is
\begin{equation}
  \label{eq:H-pole1}
  \frac{1}{\varepsilon}\!\left(
  -\frac{11}{6}C_A-3C_F+\frac{n_f}{3}
  +(C_A-2C_F)\ln\frac{\mu_R^2}{Q_{12}^2}
  -C_A\ln\frac{\mu_R^2}{Q_{13}^2}
  -C_A\ln\frac{\mu_R^2}{Q_{23}^2}
  \right)\,.
\end{equation}

\paragraph{Comparison with Ref.~\cite{Ravindran:2002dc}}




Now comparing with Ref.~\cite{Ravindran:2002dc}. The bare cross section at leading-order is
\begin{equation}
  s^2\frac{d^2\sigma_{q\bar{q}\to g H}^{(1)}}{dt\,du} =
  \pi\delta(s+t+u-M^2)\,
  \frac{e^{\varepsilon\gamma_E}}{\Gamma(1-\varepsilon)}
  S_\varepsilon\,\frac{G^2}{4N^2}
  \left(\frac{\mu_0^2\,s}{tu}\right)^\varepsilon
  \frac{\alpha_s(\mu_0^2)}{4\pi}|M_{q\bar{q}\to g H}^{(1)}|^2
  \,,
\end{equation}
where $N=N_c$ denotes the number of colours, $S_\varepsilon=\exp(\varepsilon(\ln4\pi-\gamma_E))$,
$G$ is an effective coupling where for the scenario we are considering we can approximate this to be $G\sim\alpha_s(\mu_0)/4\pi$\footnote{Please note that we believe there to be a mistake in
their equation, where they write $\alpha_s(\mu_R)$ in $G$.}, and $\mu_0$ is the bare reference scale.  The bare one-loop virtual correction takes the form
\begin{equation}
\label{eq:VIRT-bare}
\begin{aligned}
  &s^2\frac{d^2\hat\sigma_{q\bar{q}\to g H}^{\rm VIRT}}{dt\,du} =
  \pi\delta(Q_{12}^2-Q_{13}^2-Q_{23}^2-M^2)\,S_\varepsilon^2\,\frac{G^2}{N^2}
  \left(\frac{\alpha_s(\mu_0)}{4\pi}\right)^2 \times \\
  &\qquad\qquad\left[ \left\{ n_f\!\left(\frac{2}{3\varepsilon}
  +\frac{1}{3}\ln\frac{Q_{13}^2}{\mu_0^2}
  +\frac{1}{3}\ln\frac{Q_{23}^2}{\mu_0^2}
  -\frac{5}{9}\right) \right. \right.\\
   &\qquad\qquad+C_A\!\left(-\frac{2}{\varepsilon^2}
  +\!\left(2\ln\frac{Q_{12}^2}{\mu_0^2}
  -2\ln\frac{Q_{13}^2}{\mu_0^2}
  -2\ln\frac{Q_{23}^2}{\mu_0^2}
  -\frac{11}{3}\right)\frac{1}{\varepsilon}\right.\\
  &\qquad\qquad+\operatorname{Li}_2\!\left(
  \frac{Q_{12}^2-M^2}{Q_{12}^2}\right)
  -\ln\frac{Q_{13}^2}{\mu_0^2}\ln\frac{Q_{23}^2}{\mu_0^2}
  +\ln\frac{Q_{13}^2}{\mu_0^2}\ln\frac{Q_{12}^2}{\mu_0^2}
  +\ln\frac{Q_{23}^2}{\mu_0^2}\ln\frac{Q_{12}^2}{\mu_0^2}\\
  &\qquad\qquad\left.-\ln^2\frac{Q_{13}^2}{\mu_0^2}
  -\ln^2\frac{Q_{23}^2}{\mu_0^2}
  -\frac{11}{6}\ln\frac{Q_{13}^2}{\mu_0^2}
  -\frac{11}{6}\ln\frac{Q_{23}^2}{\mu_0^2}
  -\frac{7}{2}\zeta_2+\frac{38}{9}\right)\\
  &\qquad\qquad+C_F\!\left(-\frac{4}{\varepsilon^2}
  +\!\left(-2\ln\frac{Q_{13}^2}{\mu_0^2}
  -2\ln\frac{Q_{23}^2}{\mu_0^2}+3\right)\frac{1}{\varepsilon}
  +\operatorname{Li}_2\!\left(\frac{-Q_{13}^2}{M^2}\right)
  +\operatorname{Li}_2\!\left(\frac{-Q_{23}^2}{M^2}\right)\right.\\
  &\qquad\qquad-2\ln\frac{Q_{13}^2}{\mu_0^2}\ln\frac{Q_{23}^2}{\mu_0^2}
  -\frac{1}{2}\ln^2\!\left(\frac{Q_{13}^2+M^2}{Q_{13}^2}\right)
  -\frac{1}{2}\ln^2\!\left(\frac{Q_{23}^2+M^2}{Q_{23}^2}\right)\\
  &\qquad\qquad\left.+\frac{1}{2}\ln^2\!\left(\frac{M^2+Q_{13}^2}{M^2}\right)
  +\frac{1}{2}\ln^2\!\left(\frac{M^2+Q_{23}^2}{M^2}\right)
  +\frac{3}{2}\ln\frac{Q_{13}^2}{\mu_0^2}
  +\frac{3}{2}\ln\frac{Q_{23}^2}{\mu_0^2}
  +9\zeta_2-4\right)\\
  &\qquad\qquad\left.+\frac{3}{2}\ln\frac{Q_{13}^2}{\mu_0^2}
  +\frac{3}{2}\ln\frac{Q_{23}^2}{\mu_0^2}
  +9\zeta_2-4\right\}|M^{(1)}_{q\bar{q}\to g H}|^2 \\
  &\qquad\qquad\left. +\frac{(C_A-C_F)}{2}C_A C_F\left(Q_{13}^2+Q_{23}^2\right)\right]\,.
\end{aligned}
\end{equation}

We now renormalise the coupling using the Catani
convention~\cite{Catani:1998bh},
\begin{equation}
  \label{eq:Catani-renorm}
  \alpha_s(\mu_0^2) =
  \frac{\alpha_s(\mu_R^2)}{S_\varepsilon}
  \left(\frac{\mu_R^2}{\mu_0^2}\right)^\varepsilon
  \!\left[1-\frac{\alpha_s(\mu_R^2)\beta_0}{\varepsilon}
  +\mathcal{O}(\alpha_s^2)\right]\,.
\end{equation}

The bare LO cross section
contains three powers of $\alpha_s(\mu_0^2)$ (two inside $G^2$
and one explicit) and so renormalising each gives $-3\beta_0/\varepsilon$
from the explicit couplings:
\begin{equation}
\begin{aligned}
  \label{eq:LO-ren}
  s^2\frac{d^2\sigma_{q\bar{q}\to gH}^{{\rm LO},\rm Ren}}{dt\,du}& =
  \pi\delta(Q_{12}^2-Q_{13}^2-Q_{23}^2-M^2)\,
  \frac{e^{\varepsilon\gamma_E}}{\Gamma(1-\varepsilon)}\,
  \frac{G_1^2}{4N^2}
  \left(\frac{\mu_R^2\,s}{tu}\right)^\varepsilon
  \left(\frac{\alpha_s(\mu_R^2)}{4\pi}\right)^3\times \\
&\times  \!\left[1-\frac{3\alpha_s(\mu_R^2)\beta_0}{\varepsilon}
  +\mathcal{O}(\alpha_s^2)\right] |M_{q\bar{q}\to gH}^{(0)}|^2\,,
  \end{aligned}
\end{equation}
where $G_1$ denotes $G$ with $\alpha_s(\mu_0^2)$ replaced by
$\alpha_s(\mu_R^2)$.  \ABquestion{I think that $G_1$
could actually go?}
The renormalised virtual term is
\begin{equation}
\label{eq:virt-ren}
\begin{aligned}
  &s^2\frac{d^2\hat\sigma_{q\bar{q}\to g H}^{\rm VIRT, Ren}}{dt\,du} =
  \pi\delta(Q_{12}^2-Q_{13}^2-Q_{23}^2-M^2)\,\,\frac{G_1^2}{N^2}
  \left(\frac{\alpha_s(\mu_R)}{4\pi}\right)^4\left(\frac{\mu_R^2}{\mu_0^2}\right)^{2\varepsilon} \times \\
  &\qquad\qquad\left[ \left\{ n_f\!\left(\frac{2}{3\varepsilon}
  +\frac{1}{3}\ln\frac{Q_{13}^2}{\mu_0^2}
  +\frac{1}{3}\ln\frac{Q_{23}^2}{\mu_0^2}
  -\frac{5}{9}\right) \right. \right.\\
   &\qquad\qquad+C_A\!\left(-\frac{2}{\varepsilon^2}
  +\!\left(2\ln\frac{Q_{12}^2}{\mu_0^2}
  -2\ln\frac{Q_{13}^2}{\mu_0^2}
  -2\ln\frac{Q_{23}^2}{\mu_0^2}
  -\frac{11}{3}\right)\frac{1}{\varepsilon}\right.\\
  &\qquad\qquad+\operatorname{Li}_2\!\left(
  \frac{Q_{12}^2-M^2}{Q_{12}^2}\right)
  -\ln\frac{Q_{13}^2}{\mu_0^2}\ln\frac{Q_{23}^2}{\mu_0^2}
  +\ln\frac{Q_{13}^2}{\mu_0^2}\ln\frac{Q_{12}^2}{\mu_0^2}
  +\ln\frac{Q_{23}^2}{\mu_0^2}\ln\frac{Q_{12}^2}{\mu_0^2}\\
  &\qquad\qquad\left.-\ln^2\frac{Q_{13}^2}{\mu_0^2}
  -\ln^2\frac{Q_{23}^2}{\mu_0^2}
  -\frac{11}{6}\ln\frac{Q_{13}^2}{\mu_0^2}
  -\frac{11}{6}\ln\frac{Q_{23}^2}{\mu_0^2}
  -\frac{7}{2}\zeta_2+\frac{38}{9}\right)\\
  &\qquad\qquad+C_F\!\left(-\frac{4}{\varepsilon^2}
  +\!\left(-2\ln\frac{Q_{13}^2}{\mu_0^2}
  -2\ln\frac{Q_{23}^2}{\mu_0^2}+3\right)\frac{1}{\varepsilon}
  +\operatorname{Li}_2\!\left(\frac{-Q_{13}^2}{M^2}\right)
  +\operatorname{Li}_2\!\left(\frac{-Q_{23}^2}{M^2}\right)\right.\\
  &\qquad\qquad-2\ln\frac{Q_{13}^2}{\mu_0^2}\ln\frac{Q_{23}^2}{\mu_0^2}
  -\frac{1}{2}\ln^2\!\left(\frac{Q_{13}^2+M^2}{Q_{13}^2}\right)
  -\frac{1}{2}\ln^2\!\left(\frac{Q_{23}^2+M^2}{Q_{23}^2}\right)\\
  &\qquad\qquad\left.+\frac{1}{2}\ln^2\!\left(\frac{M^2+Q_{13}^2}{M^2}\right)
  +\frac{1}{2}\ln^2\!\left(\frac{M^2+Q_{23}^2}{M^2}\right)
  +\frac{3}{2}\ln\frac{Q_{13}^2}{\mu_0^2}
  +\frac{3}{2}\ln\frac{Q_{23}^2}{\mu_0^2}
  +9\zeta_2-4\right)\\
  &\qquad\qquad\left.+\frac{3}{2}\ln\frac{Q_{13}^2}{\mu_0^2}
  +\frac{3}{2}\ln\frac{Q_{23}^2}{\mu_0^2}
  +9\zeta_2-4\right\}|M^{(1)}_{q\bar{q}\to g H}|^2 \\
  &\qquad\qquad\left. +\frac{(C_A-C_F)}{2}C_A C_F\left(Q_{13}^2+Q_{23}^2\right)\right]\,.
\end{aligned}
\end{equation}
The relevant combination for extracting the hard function is the
sum of the renormalised LO and virtual terms. To add them, both must
be expressed at the same order $(\alpha_s/4\pi)^4$. The LO term at
order $(\alpha_s/4\pi)^3$ acquires one additional power from the NLO
expansion, which generates a $-3\beta_0/\varepsilon$ from renormalising
the three bare couplings at the NLO level. Combined with the virtual
term:
\begin{multline}
  s^2\frac{d^2\sigma_{q\bar{q}\to g H}^{(1),\rm Ren}}{dt\,du}
  +s^2\frac{d^2\hat\sigma_{q\bar{q}\to g H}^{\rm VIRT,Ren}}{dt\,du}=\\
  \simeq\pi\delta(Q_{12}^2-Q_{13}^2-Q_{23}^2-M^2)\,
  \frac{e^{\varepsilon\gamma_E}}{\Gamma(1-\varepsilon)}\,
  \frac{G_1^2}{4N^2}
  \left(\frac{\mu_R^2\,s}{tu}\right)^\varepsilon
  \left(\frac{\alpha_s(\mu_R^2)}{4\pi}\right)^4\\
  \times\left[-\frac{3\beta_0}{\varepsilon}
  +4\left(\frac{\mu_R^2}{\mu_0^2}\right)^\varepsilon
  \left(\frac{tu}{\mu_0^2\,s}\right)^\varepsilon
  \times[\dots]\right]|M^{(1)}_{q\bar{q}\to g H}|^2\,,
\end{multline}
where $\dots$ are the terms within the curly brackets in
Eq.~\eqref{eq:virt-ren}. 

Note that in Ref.~\cite{Ravindran:2002dc} the
$e^{\varepsilon\gamma_E}/\Gamma(1-\varepsilon)$ prefactor has been
fully expanded inside the bracket rather than appearing as an overall
factor. When extracting the poles and finite terms, this prefactor
must be divided out to match the ARES/Catani convention of
eq.~\eqref{eq:HQ-expanded}. Using the expansion
\begin{equation}
  \frac{e^{\varepsilon\gamma_E}}{\Gamma(1-\varepsilon)} =
  1 - \frac{\zeta_2}{2}\varepsilon^2 + \mathcal{O}(\varepsilon^3)\,,
\end{equation}
and combining with the double-poles we find this simply subtracts the
$\zeta_2$ terms present within the formula above.

\paragraph{Pole comparison}
Reading off the $1/\varepsilon^2$ and $1/\varepsilon$ coefficients
from Eq.~\eqref{eq:VIRT-bare}, the double-pole coefficient is
\begin{equation}
  \label{eq:vN-pole2}
  -\frac{2}{\varepsilon^2}(C_A+2C_F)\,,
\end{equation}
and the single-pole coefficient is
\begin{equation}
  \label{eq:vN-pole1}
  \frac{1}{\varepsilon}\!\left(
  -\frac{11}{3}C_A-6C_F+\frac{2n_f}{3}
  -2C_A\ln\frac{\mu_R^2}{Q_{12}^2}
  +2(C_A+C_F)\ln\frac{\mu_R^2}{Q_{13}^2}
  +2(C_A+C_F)\ln\frac{\mu_R^2}{Q_{23}^2}
  \right)\,.
\end{equation}
Comparing with Eqs.~\eqref{eq:H-pole2} and~\eqref{eq:H-pole1},
the poles of Ref.~\cite{Ravindran:2002dc} are exactly twice those
of $\mathcal{V}(Q;\varepsilon)$, this factor is taken into account when 
subtracting the finite terms.

\paragraph{Extraction of the finite remainder}
The finite remainder is extracted from the
$\mathcal{O}(\varepsilon^0)$ terms of eq.~\eqref{eq:VIRT-bare}
after subtracting the predicted poles and dividing out the
$e^{\varepsilon\gamma_E}/\Gamma(1-\varepsilon)$ prefactor, giving rise to
the SCET hard function
\begin{equation}
\label{eq:C1-SCET}
\begin{aligned}
& \frac{\left(\vec{C}^{\dagger(1)}\vec{C}^{(0)}+\vec{C}^{\dagger(0)}\vec{C}^{(1)}\right)}{|\vec{C}^{(0)}|^2}=\;\\
& C_A\left[
\frac{152}{9}
+4\operatorname{Li}_2\left(\frac{Q_{12}^2-M^2}{Q_{12}^2}\right)
-\frac{22}{3}\ln\frac{Q_{12}^2}{\mu_R^2}
+\ln^2\frac{\mu_R^2}{Q_{12}^2}
-\ln^2\frac{\mu_R^2}{Q_{13}^2}
-\ln^2\frac{\mu_R^2}{Q_{23}^2}
+2\ln\frac{Q_{12}^2}{Q_{13}^2}
 \ln\frac{Q_{12}^2}{Q_{23}^2}
\right]\\
&+C_F\left[-16
+4\operatorname{Li}_2\left(\frac{-Q_{13}^2}{M^2}\right)
+4\operatorname{Li}_2\left(\frac{-Q_{23}^2}{M^2}\right)
-2\ln^2\frac{-Q_{13}^2-M^2}{-Q_{13}^2}
-2\ln^2\frac{-Q_{23}^2-M^2}{-Q_{23}^2}\right.\\
&\qquad\left.
+2\ln^2\frac{M^2+Q_{13}^2}{M^2}
+2\ln^2\frac{M^2+Q_{23}^2}{M^2}
-2\ln^2\frac{Q_{12}^2}{\mu_R^2}
+6\ln\frac{Q_{12}^2}{\mu_R^2}
+2\ln^2\frac{Q_{13}^2}{Q_{23}^2}
\right]\\
&+n_f\left(-\frac{20}{9}-\frac{4}{3}\ln\frac{\mu_R^2}{Q_{12}^2}\right)+\pi^2\left(
6C_F-\frac{7}{3}C_A
\right)\,.
\end{aligned}
\end{equation}

For the \textsc{ARES} formalism, the hard coefficient is given by
Eq.~\eqref{eq:H1-final},
\begin{equation}
H_{q\bar{q}}^{(1)}=\mathcal{H}_1
+3\left(C_F-\frac{C_A}{2}\right)\ln\frac{Q_{12}^2}{Q^2}
+\left(\frac{3C_A}{4}+\pi\beta_0\right)\ln\frac{Q_{13}^2}{Q^2}
+\left(\frac{3C_A}{4}+\pi\beta_0\right)\ln\frac{Q_{23}^2}{Q^2}\,.
\end{equation}

Subtracting these from
Eq.~\eqref{eq:C1-SCET} and accounting for the fact that there is a factor of 2 arising from
\textsc{ARES} expressing $\alpha_s/2\pi$
opposed to $\alpha_s/4\pi$, we obtain $\mathcal{H}_1$:
\begin{equation}
\begin{aligned}
\mathcal{H}_1=&-\frac{5}{4}C_A\pi^2+\frac{17}{6}C_F\pi^2
+\frac{3}{2}C_A\ln\frac{Q_{12}^4}{Q_{13}^2Q_{23}^2}
-\pi\beta_0\ln\frac{Q_{12}^8Q_{13}^2Q_{23}^2}{\mu_R^{12}}
\\
&+\frac{C_A}{2}\left[
\frac{152}{9}
+4\operatorname{Li}_2\left(\frac{Q_{12}^2-M^2}{Q_{12}^2}\right)
+2\ln\frac{Q_{12}^2}{Q_{13}^2}
 \ln\frac{Q_{12}^2}{Q_{23}^2}
\right]
\\
&+\frac{C_F}{2}\left[
-16
+4\operatorname{Li}_2\left(\frac{-Q_{13}^2}{M^2}\right)
+4\operatorname{Li}_2\left(\frac{-Q_{23}^2}{M^2}\right)
-2\ln^2\frac{-Q_{13}^2-M^2}{-Q_{13}^2}
-2\ln^2\frac{-Q_{23}^2-M^2}{-Q_{23}^2}
\right.
\\
&\left.
+2\ln^2\frac{M^2+Q_{13}^2}{M^2}
+2\ln^2\frac{M^2+Q_{23}^2}{M^2}
+2\ln^2\frac{Q_{13}^2}{Q_{23}^2}
\right]
\\
&-\frac{20}{18}n_f\,.
\end{aligned}
\end{equation}
