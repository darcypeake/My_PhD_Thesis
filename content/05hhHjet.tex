\section{Introduction}
Here we present original research on the \textsc{ARES} method for $hh\to Z+\mathrm{jet}$ production.
We extend the ARES nethod for next-to-next-to-leading
  logarithmic (NNLL) resummations to hadron collisions, considering
  the test-case scenario of event shapes in events with a $Z$ boson
  recoiling against a jet with high-transverse momentum. We
  concentrate on the example of out-of-plane radiation, and first
  compute NNLL contributions arising from soft radiation, as well as
  hard collinear emissions from the final state. These can be obtained
  with the same formulae developed for three-jet events in $e^+e^-$
  annihilation. We then improve the ARES framework to include the
  effect of initial-state radiation at NNLL accuracy
  %, perform some basic phenomenology 
  and give a general formula for an arbitrary
  observable in events with a massive boson and a jet.

\section{Proof-of-concept observables in $Z$+jet events}
\label{sec:observable}

We consider the production of a $Z$ boson and a hard jet in hadron collisions,
\begin{equation}
  \label{eq:Z+jet}
  h_1(P_1)+h_2(P_2) \to Z(q) + \mathrm{jet}(p_J)+\dots\,,
\end{equation}
where $h_1, h_2$ are the two incoming hadrons, and the dots represents
accompanying hadrons. At Born level, such events results from two
partons of momenta $p_1=x_1 P_1, p_2=x_2 P_2$ producing a $Z$ boson
$q$ and a hard parton $p_3$. Beyond Born level, $p_3$ will recoil
against additional partons. Its momentum will change to
$\tilde p_3$, and it will be clustered with other partons to form a hard jet $p_J$. At Born level, $p_J=p_3$. 

We set the $z$-axis along the direction of the beam, the $y$-axis
along the direction of the transverse momentum of the $Z$ boson. There
are various possibilities to define observables that measure radiation
out of the plane identified by the beam and the $Z$ boson.

\begin{itemize}
\item \textbf{One-jettiness} is a special case of the more general $N$-jettiness event shape~\cite{Stewart:2010tn}, which quantifies how well the final state can be described by $N$ narrow jets. For $N=1$, one-jettiness is defined as
\begin{equation}
 \label{eq:T1}
 \mathcal{T}_1\equiv \sum_{i} \min\left\{\frac{2 P_1\cdot q_i}{Q_1}, \frac{2 P_2\cdot q_i}{Q_2}, \frac{2 p_J \cdot q_i}{Q_3}\right\}\,,
\end{equation}
where $q_i$ is the four-momentum of a \emph{hadron}, and
$Q_1, Q_2, Q_3$ three momentum scales.  When $\mathcal{T}_1\to 0$, the
outgoing jet $p_J$ recoils against the $Z$ boson. We can also consider a normalised version of 1-jettiness,
$\tau_1\equiv \mathcal{T}_1/M$, with $M$ a momentum-scale normalisation.
A particular choice of $Q_1, Q_2, Q_3$
\begin{equation}
  \label{eq:Qi-inv}
  Q_1 = \frac{2 P_1\cdot p_J}{p_{T,J}}\,,  \quad Q_2 = \frac{2 P_2\cdot p_J}{p_{T,J}}\,,
  \quad Q_3 = 2 p_{T,J}\,,
\end{equation}
where $p_{T,J}$ is the transverse momentum of the outgoing jet with
respect to the beam, provides a definition of $\mathcal{T}_1$ that is
manifestly invariant under boosts along the beam direction:
\begin{equation}
	\label{eq:T1-binv}
	\mathcal{T}_1\equiv  \sum_{i} k_{t,i}\min\{e^{(\eta_i-\eta_J)}, e^{-(\eta_i-\eta_J)}, \cosh(\eta_{i}-\eta_J)-\cos(\phi_{i}-\phi_J)\}\,,
\end{equation}
which is of course independent of the two incoming momenta. To reduce
the impact of experimental systematics in the measurement of particle momenta, we
set $M=H_T$, with $H_T$ the scalar sum of all
hadron transverse momenta, leading to our preferred definition of $\tau_1$, namely
\begin{equation}
	\label{eq:tau1-binv}
	\tau_1\equiv  \sum_{i} \frac{k_{t,i}}{H_T}\min\{e^{(\eta_i-\eta_J)}, e^{-(\eta_i-\eta_J)}, \cosh(\eta_{i}-\eta_J)-\cos(\phi_{i}-\phi_J)\}\,.
\end{equation}

The normalisation factors are $Q_i = 2 \rho_i E_i$, with $\rho_i$ depending on the reference frame (e.g., $\rho_i = 1$ in the laboratory frame). While $\mathcal{T}_1$ is dimensionful, it is often convenient to define the dimensionless variable $\tau_1 \equiv \mathcal{T}_1 / Q$, where $Q$ is the hard scale of the interaction.

\item \textbf{Transverse thrust}, first introduced in Ref.~\cite{Banfi:2004nk}, is the extension of the $e^+e^-$ event-shape thrust (cf.\ in Eq.~\eqref{eq:thrust}) to hadronic collisions. It is defined in the transverse plane as
\begin{equation}
\label{eq:tperp-def}
  T_t \equiv \max_{\vec{n}_\perp}
  \frac{\sum_i \left| \vec{q}_{\perp i} \cdot \vec{n}_\perp \right|}
       {\sum_i \left| \vec{q}_{\perp i} \right|},
  \qquad
  \tau_t \equiv 1 - T_\perp \,,
\end{equation}
where $\vec{p}_{\perp i}$ denotes the transverse momentum of final-state particle $i$ with respect to the beam direction, and the unit vector $\vec{n}_\perp$ that maximises this quantity is referred to as the \emph{transverse thrust axis}. This variable is invariant
under longitudinal boosts, and, in the present case, can be defined in
terms of hadron transverse momenta $\vec q_{t,i}$ and the $Z$-boson
transverse momentum $\vec q_T$ as follows:
\begin{equation}
  \label{eq:tthr}
  T_t\equiv \frac{\sum_i |\vec q_{t,i}\cdot \vec q_T|}{ H_T q_T}\,.
\end{equation}
In particular, the near-to-planar region corresponds to $\tau_t\equiv 1-T_t$ close to zero. Both $\tau_1$ and $\tau_t$ are sensitive to out-of-plane radiation through imbalance between large in-plane momentum components. 

\item \textbf{Thrust minor}, also introduced in Ref.~\cite{Banfi:2004nk}, is a measure of out-of-plane radiation and is defined as
\begin{equation}
\label{eq:Tmin-def}
  T_m \equiv \frac{\sum_i |\vec q_{\perp i}\times \vec n_\perp|}{\sum_i q_{\perp i}}\,.
\end{equation}
For the present case, this can be written as
\begin{equation}
  \label{eq:Tmin}
  T_m \equiv \sum_i\frac{ |\vec q_{t,i}\times \vec q_T|}{H_T q_T}\,.
\end{equation}

The one-jettiness, one minus the transverse thrust and the thrust
minor are all examples of rIRC safe observables. These are generic
functions $V(\{\tilde p\},k_1,\dots,k_n)$ of final-state momenta
$\{\tilde p\}\equiv\{\tilde p_1,\tilde p_2,\tilde p_3\}$, accompanied
by additional emissions $k_1,\dots,k_n$. Any rIRC safe observable $V$
is zero at Born level, i.e.\
$V(\{\tilde p_1,\tilde p_2,\tilde p_3\}) = V(\{p_1,p_2,p_3\})=0$, and
satisfies additional conditions on its scaling with an arbitrary
number of soft and/or collinear emissions~\cite{Banfi:2004yd}.  As
such, all the aforementioned observables fall within the scope of the
CAESAR/ARES formalism.

The starting point is to consider the behaviour of each observable in the presence of a single soft emission collinear to leg $\ell$. \begin{itemize}

\item{\textbf{One-jettiness.}} By definition,
  $k\cdot p_\ell/E_\ell=k_t^{(\ell)} e^{-\eta^{(\ell)}}$, where
  $k_t^{(\ell)}$ is the relative transverse momentum with respect to
  the emitting leg $\ell$, $\eta^{(\ell)}$ is the rapidity of the

  \begin{equation}
    \label{eq:tau1-sc-1emsn}
    \tau_1(\{\tilde p\},k)\simeq \tau^{\mathrm{sc}}_1(\{\tilde p\},k) =  \frac{2E_\ell}{x_\ell Q_\ell}\frac{k_t^{(\ell)}}{M} e^{-\eta^{(\ell)}}\,.
  \end{equation}
Using the explicit expression for the scales $Q_1,Q_2,Q_3$ in eq.~(\ref{eq:Qi-inv}), as well as the fact that, for soft emissions, $H_T\simeq q_T$ and $\eta_J\simeq \eta_3$, we obtain
  \begin{equation}
\label{eq:tau1-sc-1emsn}
  \tau_1(\{\tilde p\},k) \simeq \tau^{\mathrm{sc}}_1(\{\tilde p\},k) = \frac{k_t^{(\ell)}}{q_T}e^{-\eta^{(\ell)}}\times \left\{
    \begin{split}
    & e^{\eta_3}\,,\qquad\qquad \ell=1\\
    & e^{-\eta_3}\,, \qquad\quad\> \ell = 2 \\
      & \cosh\eta_3 \,,\qquad \ell=3\,.
    \end{split}
  \right. 
\end{equation}

\item {\textbf{Transverse thrust.}} This case is the most complicated. First we observe that, in the presence of a single soft emission $k$, we have
\begin{equation}
\label{eq: taut-1em}
  \begin{split}
  \tau_t(\{\tilde p\},k)& \simeq  \frac{k_t-|k_y|+\sqrt{\tilde p_{3y}^2+\tilde p_{3x}^2}-|\tilde p_{3y}|}{q_T}= \frac{k_t-|k_y|}{q_T}+\mathcal{O}\left(\frac{k_x^2}{q_T^2}\right)\,.
\end{split}
\end{equation}
This gives a different behaviour according to the emitting leg:
\begin{equation}
  \label{eq:taut-sc-1emsn}
  \tau_t(\{\tilde p\},k) \simeq \tau^{\mathrm{sc}}_t(\{\tilde p\},k) = \frac{k_t^{(\ell)}}{q_T}\times \left\{
    \begin{split}
    & \left(1-|\cos\phi^{(\ell)}|\right) \,,\quad \ell=1,2\\
      & \frac{E_\ell}{q_T}e^{-\eta^{(\ell)}} \sin^2\phi^{(\ell)}\,,\qquad \ell=3\,.
    \end{split}
  \right. 
\end{equation}
\item{\textbf{Thrust minor.}} In this case, we need the out of plane
  component of transverse momentum of emission $k$. Setting the
  $y$-axis along the direction of the transverse momentum of the $Z$
  boson, we call this component $k_x$. We then introduce
  $\phi^{(\ell)}$, the azimuthal angle of the transverse momentum of
  emission $k$ with respect to leg $\ell$. Setting by convention
  $\phi^{(\ell)}=0$ when the emission is in the event plane gives
  $k_x\equiv k_t^{(\ell)} \sin\phi^{(\ell)}$ for any emitting leg
  $\ell$. With these kinematical variables, we obtain
  \begin{equation}
  \begin{split}
  T_m(\{\tilde p\},k)\simeq  \frac{|k_x|+|\tilde p_{3x}|}{q_T} \,.
\end{split}
\end{equation}
Using transverse momentum conservation $\tilde p_{3x}+k_x=0$, we arrive at
our final expression
\begin{equation}
  \label{eq:Tm-sc-1emsn}
  \begin{split}
  T_m(\{\tilde p\},k)& \simeq  T^{\mathrm{sc}}_m(\{\tilde p\},k) = \frac{2|k_x|}{q_T} = 2 \frac{k_t^{(\ell)}}{q_T} |\sin\phi^{(\ell)}| \,.
\end{split}
\end{equation}
In the presence of
multiple soft and collinear emissions $k_1, \dots, k_n$, eq.~(\ref{eq:Tmin}) can be
approximated as follows:
\begin{equation}
  \label{eq:Tm-sc}
  \begin{split}
  T_m(\{\tilde p\},k_1,\dots,k_n) & \simeq \frac{1}{q_T} \left(\sum_{i=1}^n |k_{ix}| + |\tilde p_{3x}|\right)\\ & =\frac{1}{q_T} \left(\sum_{i=1}^n |k_{ix}| + \left|\sum_{i=1}^n k_{ix}\right|\right)\equiv T^{\mathrm{sc}}_m(\{\tilde p\},k_1,\dots,k_n)\,,
  \end{split}
\end{equation}
where we have used the fact that, in the presence of soft emissions,
$p_{\mathrm{t,jet}}\simeq q_T$, and that $\tilde p_{3x}$ can be obtained
from conservation of transverse momentum. For a soft emission $k$
collinear to leg $\ell$, $T^{\mathrm{sc}}_m(\{\tilde p\},k)$ is given by
eq.~(\ref{eq:Tm-sc-1emsn}).
