\section{Proof-of-concept observables in $Z$+jet events}
\label{sec:observable}

We consider the production of a $Z$ boson and a hard jet in hadron collisions,
\begin{equation}
  \label{eq:Z+jet}
  h_1(P_1)+h_2(P_2) \to Z(q) + \mathrm{jet}(p_J)+\dots\,,
\end{equation}
where $h_1, h_2$ are the two incoming hadrons, and the dots represents
accompanying hadrons. At Born level, such events results from two
partons of momenta $p_1=x_1 P_1, p_2=x_2 P_2$ producing a $Z$ boson
$q$ and a hard parton $p_3$. Beyond Born level, $p_3$ will recoil
against additional partons. Its momentum will change to
$\tilde p_3$, and it will be clustered with other partons to form a hard jet $p_J$. At Born level, $p_J=p_3$. \\ \\
We set the $z$-axis along the direction of the beam, the $y$-axis
along the direction of the transverse momentum of the $Z$ boson. There
are various possibilities to define observables that measure radiation
out of the plane identified by the beam and the $Z$ boson.

\section{NNLL resummation: soft and final-state hard-collinear radiation}
\label{sec:NNLL-final}
The one-jettiness, one minus the transverse and the thrust thrust
minor are all examples of rIRC safe observables. These are generic
functions $V(\{\tilde p\},k_1,\dots,k_n)$ of final-state momenta
$\{\tilde p\}\equiv\{\tilde p_1,\tilde p_2,\tilde p_3\}$, accompanied
by additional emissions $k_1,\dots,k_n$. Any rIRC safe observable $V$
is zero at Born level, i.e.\
$V(\{\tilde p_1,\tilde p_2,\tilde p_3\}) = V(\{p_1,p_2,p_3\})=0$, and
satisfies additional conditions on its scaling with an arbitrary
number of soft and/or collinear emissions~\cite{Banfi:2004yd}.  As
such, all the aforementioned observables fall within the scope of the

\section{Initial State Radiation}
In order to extend ARES to initial-state radiation, we follow the same principles as for final-state radiation, namely:
\begin{itemize}
\item we add one collinear emission $k$, and the corresponding virtual
  corrections, on top of hard Born partons, accompanied by an arbitrary
  number of soft and collinear emissions $\{k_i\}$;
\item we subtract the double counting occurring with the soft and collinear region.
\end{itemize}
We use dimensional regularisation to make each contribution
individually finite. In the case of initial-state radiation, virtual
corrections include the counterterm arising from the renormalisation
of PDFs. \\ \\
To be concrete, we consider the case of an incoming quark carrying a fraction $x/z$ of the proton momentum. This emits a
collinear gluon carrying a fraction $1-z$ of its emitter's momentum. After the emission, the quark enters the hard scattering with a
fraction $x$ of the proton momentum. The above procedure gives a
function that we call $\mathcal{F}_{\mathrm{ISR}}$. This function will be
reorganised to give various NNLL functions. Our starting point is
\begin{equation}
  \label{eq:FISR-start}
  \begin{split}
    \mathcal{F}_{\mathrm{ISR}} & = \int \dZ \left\{\int_0^{Q^2}\frac{dk_t^2}{k_t^2}\left(\frac{\mu_R^2}{k_t^2}\right)^{\varepsilon} \frac{\alpha_s(k_t)}{2\pi}\times\right. \\& \times \left[\int_x^1 \frac{dz}{z}C_F \left[\frac{1+z^2}{1-z}-\varepsilon(1-z)\right]\Theta\left(1-\frac{V_{\mathrm{hc}}(\{\tilde{p}\},k,\{k_i\}}{v}\right) \frac{f_q(x/z,\mu_F)}{f_q(x,\mu_F)} + \right. \\
    & \left.- \int_0^1 dz \frac{2C_F}{1-z}\Theta\left(1-\frac{V_{\mathrm{sc}}(\{\tilde{p}\},k,\{k_i\}}{v}\right)\right] + \\
   & + \left[-\gamma_q^{(0)}\int_0^{Q^2v^{2/(a+b_q)}}\frac{dk^2}{k^2}\left(\frac{\mu_R^2}{k^2}\right)^{\varepsilon}\frac{\alpha_s(k)}{2\pi}
   +\frac{1}{\varepsilon} \left(\frac{\mu_R^2}{\mu_F^2}\right)^{\varepsilon} \frac{\alpha_s(\mu_R)}{2\pi} \int_x^1 \frac{dz}{z} C_F\left(\frac{1+z^2}{1-z}\right)_+ \frac{f_q(x/z,\mu_F)}{f_q(x,\mu_F)}\right] \times \\ & \times \left. \Theta\left(1-\frac{V_{\mathrm{sc}}(\{\tilde{p}\},\{k_i\}}{v}\right)\right\} \,.
\end{split}
\end{equation}

\begin{equation}
\begin{aligned}
\delta \mathcal{F}_{\mathrm{rec,q}}& =\int \dZ \frac{\alpha_s\left(Q v^{1/(a+b_\ell)}\right)}{\alpha_s(Q)\left(a+b_\ell\right)}\left\{\int_0^\infty\frac{d\zeta}{\zeta}\times\right. \\
&\times P^{\mathrm{rec}}_q(z)
\left[\Theta\left(1-\frac{V_{\mathrm{hc}}(\{\tilde{p}\},k,\{k_i\})}{v}\right)-\Theta\left(1-\frac{V_{\mathrm{sc}}(\{\tilde{p}\},k,\{k_i\})}{v}\right)\right]\,,
\end{aligned}
\end{equation}
where
\begin{equation}
  P^{\mathrm{rec}}_q(z) = \int_x^1 \frac{dz}{z}\frac{1}{f_q(x,\mu_F)}\left[ f_q(x/z,\mu_F)C_F\frac{1+z^2}{1-z}+ f_g(x/z,\mu_F)T_R(z^2+(1-z)^2)\right]\,.
\end{equation}
For final-state radiation, this reduces to
\begin{equation}
	P^{\mathrm{rec}}_q(z)=\int_0^\pi\frac{d\phi^{(\ell)}}{\pi}\int_0^1dz^{(\ell)}\left(\frac{2C_\ell}{z^{(\ell)}}+\langle P_{f_\ell}(z^{(\ell)},0)\rangle\right)\,,
\end{equation}
with
\begin{equation} 
\label{eq:P}
  \langle P_{f_\ell}(z;\varepsilon) \rangle \equiv
  \begin{cases}
    C_F(-2+(1-\varepsilon)z)\,, & f_\ell = q\,,\\
    C_A(z(1-z)-2)+T_R n_f\left[1-\frac{2z(1-z)}{1-\varepsilon}\right]\,, & f_\ell = g\,.
  \end{cases}
\end{equation}
Finally, the hard-collinear correction is
\begin{equation}
\begin{aligned}
\delta \mathcal{F}_{\mathrm{hc,q}}& = \int \dZ\frac{\alpha_s\left(v^{1/(a+b_\ell)}\right)}{\alpha_s(Q)\left(a+b_\ell\right)}\int_0^\infty\frac{d\zeta}{\zeta}  \, P^{\mathrm{hc}}_q(z) \\
&\times \left[\Theta\left(1-\frac{V_{\mathrm{sc}}(\{\tilde{p}\},k,\{k_i\})}{v}\right) -\Theta(1-\zeta)\Theta\left(1-\frac{V_{\mathrm{sc}}(\{\tilde{p}\},\{k_i\})}{v}\right)\right]\,,
\end{aligned}
\end{equation}
with
\begin{equation}
P^{\mathrm{hc}}_q(z)=\int_x^1 \frac{dz}{z}\frac{1}{f_q(x,\mu_F)}\left[ f_q(x/z,\mu_F)C_F\frac{1+z^2}{(1-z)_{+}}+ f_g(x/z,\mu_F)T_R(z^2+(1-z)^2)\right]\,.
\end{equation}
In the context of final-state radiation, this becomes
\begin{equation}
	P^{\mathrm{rec}}_q(z)=\int_0^\pi\frac{d\phi^{(\ell)}}{\pi}\int_0^1dz^{(\ell)}\langle P_{f_\ell}(z^{(\ell)},0)\rangle\,.
\end{equation}

\noindent\textbf{One-jettiness} \\ 
$q\to qg$
\begin{equation}
C^{(1)}_{\mathrm{hc,q}} =
\int_x^1\frac{dz}{z}
\frac{f_q(x/z,\mu_F)}{f_q(x,\mu_F)}
\left[
C_F(1+z^2)\left(\frac{\ln(1-z)}{(1-z)}\right)_{\!+}
-\ln(d_\ell g_\ell(\phi))
\left(P_{qq}^{(0)}-C_F\frac{3}{2}\delta(1-z)\right)
+1-z
\right].
\end{equation}
$g\to q\bar{q}$
\begin{equation}
C^{(1)}_{\mathrm{hc,q}} =
\int_x^1\frac{dz}{z}
\frac{f_g(x/z,\mu_F)}{f_q(x,\mu_F)}
\left[
T_R\!\left((1-z)^2+z^2\right)\ln(1-z)
-P_{qg}^{(0)}\ln(d_\ell g_\ell(\phi))
+2T_R\,z(1-z)
\right].
\end{equation}
$q\to gq$
\begin{equation}
C^{(1)}_{\mathrm{hc,g}} =
\int_x^1\frac{dz}{z}
\frac{f_q(x/z,\mu_F)}{f_g(x,\mu_F)}
\left[
C_F\frac{1+(1-z)^2}{z}\ln(1-z)
-P_{gq}^{(0)}\ln(d_\ell g_\ell(\phi))
+C_F z
\right].
\end{equation}
\noindent\textbf{Transverse thrust and thrust minor} \\
$q\to qg$
\begin{equation}
C^{(1)}_{\mathrm{hc,q}} =
\int_x^1\frac{dz}{z}
\frac{f_q(x/z,\mu_F)}{f_q(x,\mu_F)}
\left[
-2\ln(d_\ell g_\ell(\phi))
\left(P_{qq}^{(0)}-C_F\frac{3}{2}\delta(1-z)\right)
+1-z
\right].
\end{equation}
$g\to q\bar{q}$
\begin{equation}
C^{(1)}_{\mathrm{hc,q}} =
\int_x^1\frac{dz}{z}
\frac{f_g(x/z,\mu_F)}{f_q(x,\mu_F)}
\left[
-2P_{qg}^{(0)}\ln(d_\ell g_\ell(\phi))
+2T_R\,z(1-z)
\right].
\end{equation}
$q\to gq$
\begin{equation}
C^{(1)}_{\mathrm{hc,g}} =
\int_x^1\frac{dz}{z}
\frac{f_q(x/z,\mu_F)}{f_g(x,\mu_F)}
\left[
-2P_{gq}^{(0)}\ln(d_\ell g_\ell(\phi))
+C_F z
\right].
\end{equation} \\ \\
The leading-order splitting functions are given by
\begin{align}
P_{qq}^{(0)} &= C_F\!\left[\frac{1+z^2}{(1-z)_{\!+}}+\frac{3}{2}\delta(1-z)\right], \\
P_{qg}^{(0)} &= T_R\!\left[(1-z)^2+z^2\right], \\
P_{gq}^{(0)} &= C_F\!\left[\frac{1+(1-z)^2}{z}\right].
\end{align}
