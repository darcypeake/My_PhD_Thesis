\chapter{Resummation}
In this chapter, we discuss the technique of \emph{resummation}. As outlined in Sec.~\ref{subsec:QCD-ev}, following the initial hard scattering, the produced coloured partons undergo QCD radiation dominated by soft and collinear emissions. For many observables, these emissions give rise to logarithmically enhanced contributions that become large in the limit where the observable, $v$, is small. In this regime, fixed-order perturbation theory breaks down, indicating the need for dedicated resummation techniques.

\section{Event-shapes}
\label{sec:event-shapes}
Event shapes are IRC safe observables constructed 
from the four-momenta of final-state particles, which characterise the geometry 
of an event by quantifying the distribution of energy-momentum flow in the 
hadronic final state. Unlike jet algorithms (see Sec.~\ref{subsec:jet-algorithm}), which cluster particles into 
discrete jets, event shapes provide a continuous and global description of the 
event geometry, making them sensitive to radiation across the entire phase space. 
They have been widely used in precision studies of QCD, most notably in extractions of the 
strong coupling constant $\alpha_s$~\cite{Becher:2008cf, Abbate:2010xh}. Notable examples include: 
\begin{itemize}
\item{\textbf{Thrust}}~\cite{Farhi:1977sg, Brandt:1964sa} is defined as,
\begin{equation}
\label{eq:thrust}
	T \equiv \max_{\vec{n}} \frac{\sum_j |\vec{p}_j \cdot \vec{n}|}{\sum_j |\vec{p}_j|}\,, \quad \tau \equiv 1 - T \,,
\end{equation}
where $\vec{p}_j$ denotes the three-momentum of particle $j$ in the final state, and the unit vector $\vec{n}$ that maximises this quantity is referred to as the \emph{thrust axis}. It is conventional to work with the variable $\tau$, as defined in Eq.~\eqref{eq:thrust}.

\item{\textbf{Jet Broadenings}}~\cite{Clavelli:1979md, Rakow:1981qn, Catani:1992jc} quantify the transverse momentum of final-state hadrons relative to the thrust axis by dividing the event into two hemispheres, $\mathcal{H}_L$ and $\mathcal{H}_R$, defined with respect to the thrust axis $\vec{n}_T$. For each hemisphere, the broadening is defined as
\begin{equation}
	B_\ell \equiv \frac{\sum_{i \in \mathcal{H}_\ell} |\vec{p}_i \times \vec{n}_T|}{2 \sum_i |\vec{p}_i|}, \quad \ell = L, R\,.
\end{equation}
From these, one defines the wide-jet broadening, narrow-jet broadening, and total 
broadening respectively as
\begin{equation}
    B_w \equiv \max\{B_L, B_R\}\,, \qquad 
    B_n \equiv \min\{B_L, B_R\}\,, \qquad 
    B_t \equiv B_L + B_R\,.
\end{equation}
\item{\textbf{Jet Masses}}~\cite{Chandramohan:1980ry, Clavelli:1981yh} are defined by computing the invariant mass of each 
hemisphere,
\begin{equation}
	\rho_{\ell}\equiv \frac{\left(\sum_{i\in\mathcal{H}_{\ell}}p_i\right)^2}{\left(\sum_iE_i\right)^2}\,,\quad \ell=L,R\,.
\end{equation}
The heavy-jet mass and light-jet mass are defined respectively as
\begin{equation}
	\rho_h\equiv\max\left\{\rho_L,\rho_R\right\}\,,\quad \rho_l\equiv\min\left\{\rho_L,\rho_R\right\}\,.
\end{equation}

\section{Motivation: Thrust and the Need for Resummation}
\label{sec:thrust-example}
To motivate the need for resummation, we consider a fixed-order calculation of the thrust distribution. At Born level, the process $e^+e^- \to q\bar{q}$ yields
\begin{equation}
    \frac{1}{\sigma_0}\frac{d\sigma}{d\tau} = \delta(\tau),
\end{equation}
where $\sigma_0$ denotes the leading-order cross section defined in Eq.~\eqref{eq:sigma-LO}.

The first non-trivial contribution arises from the term at $\mathcal{O}(\alpha_s)$ from the real-emission process $e^+e^- \to q\bar{q}g$. For a three-particle final state, thrust is determined by the largest momentum fraction, 
\begin{equation}
T = \max[x_q, x_{\bar{q}}, x_g],
\end{equation}
with $T_\text{min} = 2/3$ (for equal momenta) and $T_\text{max} = 1$. The corresponding differential distribution can be written as
\begin{equation}
	\frac{1}{\sigma_0}\frac{d\sigma}{dT} = C_F \frac{\alpha_s}{2\pi} \int dx_q \int dx_{\bar{q}} \frac{x_q^2 + x_{\bar{q}}^2}{(1-x_q)(1-x_{\bar{q}})} \, \delta\big(T - \max[x_q, x_{\bar{q}}, x_g]\big),
\end{equation}
\ABquestion{Same limits as in Eq.~\eqref{eq:qqg-real} i.e.\ do we need the $\Theta$-function here also?}
subject to the kinematic constraint $x_q + x_{\bar{q}} + x_g = 2$. 

Performing the phase-space integration and expressing the result in terms of $\tau \equiv 1 - T$ yields
\begin{equation}
	\frac{1}{\sigma_0}\frac{d\sigma}{d\tau} = C_F \frac{\alpha_s}{2\pi} \left\{ \frac{3(1+\tau)(3\tau-1)}{\tau} + \frac{(4+6\tau(\tau-1))\ln\frac{1-2\tau}{\tau}}{\tau(\tau-1)} \right\}.
\end{equation}

As $\tau \to 0$, the distribution develops IR divergences associated with soft and collinear emissions. These divergences are cancelled by the virtual one-loop correction to $e^+e^- \to q\bar{q}$, in direct analogy with the cancellation process discussed in Sec.~\ref{sec:divergences}, yielding a finite result as required by the IRC safety of thrust (Sec.~\ref{sec:IRC}). Using the regulated and integrated NLO cross-section given in Eq.~\eqref{eq:sigma-full}, the complete distribution can be written as 
\begin{equation}
\begin{aligned}
\label{eq:full-thrust-NLO}
    \frac{1}{\sigma_0}\frac{d\sigma}{d\tau} =\;& \delta(\tau) + C_F\frac{\alpha_s}{2\pi} \Bigg\{ \delta(\tau) \left( \frac{\pi^2}{3} - 1 \right) \\
    & + \left[ 3(1+\tau)(3\tau-1) + \frac{(4+6\tau(\tau-1))\ln(1-2\tau)}{1-\tau}\right] \left( \frac{1}{\tau} \right)_+  \\
    & - \frac{4+6\tau(\tau-1)}{1-\tau} \left( \frac{\ln \tau}{\tau} \right)_+ \Bigg\}.
\end{aligned}
\end{equation}

Neglecting the $\delta$-function contributions, the terms that become large in the $\tau \to 0$ limit are the \emph{logarithmically enhanced terms},
\begin{equation}
	\left(\frac{1}{\sigma_0}\frac{d\sigma}{d\tau}\right)_{\text{log-enh}} = C_F \frac{\alpha_s}{2\pi} \left[ -3 \left(\frac{1}{\tau}\right)_+ - 4 \left(\frac{\ln \tau}{\tau}\right)_+ \right].
\end{equation}

The cumulative distribution, or cumulant, is defined as
\begin{equation}
\label{eq:cumulant}
	\Sigma(\tau) 
	= \frac{1}{\sigma_0} \int_0^\tau d\tau' \frac{d\sigma}{d\tau'}\,.
\end{equation}
Evaluating this for the logarithmically enhanced terms gives
\begin{equation}
\label{eq:thrust-cumulant}
	\Sigma_{\text{log-enh}}(\tau) = C_F \frac{\alpha_s}{2\pi} 
	\left( -2L^2 + 3L \right),
\end{equation}
where $L \equiv \ln(1/\tau)$. All remaining terms in 
Eq.~\eqref{eq:full-thrust-NLO} remain finite as $\tau \to 0$ and do not 
contribute to the logarithmic enhancement. 

The terms in $\Sigma_{\mathrm{log\text{-}enh}}(\tau)$ diverge as 
$\tau \to 0$, signalling the breakdown of the fixed-order expansion. 
the differential distribution becomes strongly suppressed in this limit: $\tau = 0$ corresponds to an exactly back-to-back $q\bar{q}$ configuration that 
cannot occur beyond Born level. \ABquestion{ i.e. because it's NLO so you need an extra quark/gluon in the final state?}

At higher orders in perturbation theory, the cumulant receives contributions of the form 
$\alpha_s^n L^m$ with $m \leq 2n$, which grow as $\tau \to 0$ and spoil the convergence of the perturbative series in precisely the region where most of the data lies. The structure of these logarithms has a clear physical origin. Since each power of the strong coupling constant $\alpha_s$ corresponds to an additional emission, and each emission can generate up to two logarithms -- one from the soft limit and one from the collinear limit --the perturbative expansion at order $\alpha_s^n$ contains terms with up to $2n$ powers of the logarithm, i.e.\ $\alpha_s^n L^m$ with $m \leq 2n$.

To restore predictive power, the perturbative expansion must be 
reorganised and \emph{resummed}. Rather than expanding purely in powers of 
$\alpha_s$, one performs an expansion in powers of the logarithms $L$,
which becomes essential in the region $\tau \ll 1$ 
where $\alpha_s L \sim 1$. This procedure, known as \emph{resummation}, systematically accounts for these enhanced contributions to all orders in $\alpha_s$, yielding an exponentiated form for the cumulant,
\begin{equation}
	\Sigma(\tau) \simeq 
	\exp \Big\{ 
	L g_1(\alpha_s L) 
	+ g_2(\alpha_s L) 
	+ \alpha_s g_3(\alpha_s L) 
	+ \cdots 
	\Big\}.
	\label{eq:resummed-cumulant}
\end{equation}

Each function $g_i$ resum logarithms at successive logarithmic accuracies, as summarised below:

\end{tabular}
\end{center}

\section{Problem Specification}
To introduce the \textsc{ARES} formalism, we first will outline the notation and the set up including the problem specification at hand.
We consider events with $n_i$ incoming partons and an arbitrary number $N$ of final-state particles, whose momenta are denoted by $q_1, \dots, q_N$. At Born level, the process consists of $n$ hard partons (including both incoming and outgoing legs), which define the underlying hard configuration.

We define an observable $V(q_1, q_2, \dots)$ as a non-negative function of the final-state momenta. We assume that the observable is IRC safe and that it vanishes smoothly in the limit where the event approaches an $n$-jet configuration. Observables with these properties are commonly referred to as $(n+1)$-jet observables, where both the outgoing jets and the incoming beam jets are included in the counting.

To select events containing $n$ or more well-separated hard jets, we introduce a selection procedure. This can be implemented, for instance, through a jet algorithm that counts the number of jets in the event, or by imposing a cut on a suitable $n$-jet observable. We encode this requirement in a selection function $\mathcal{H}(q_1,\dots,q_N)$, defined as
\begin{equation}
\mathcal{H}(q_1,\dots,q_N)=
\begin{cases}
1\,, & \text{if the event satisfies the selection criteria},\\
0\,, & \text{otherwise}.
\end{cases}
\end{equation}

The total cross section for events passing this selection, i.e.\ the total $n$-jet cross section, is given by
\begin{equation}
	\sigma_{\mathcal{H}}=\sum_{N=n-n_i}^\infty\int d\Phi_N\frac{d\sigma_N}{d\Phi_N}\mathcal{H}(q_1,\dots,q_N)\,,
\end{equation}
where $d\sigma_N/d\Phi_N$ denotes the differential cross section for producing $N$ final-state particles, and $d\Phi_N$ is the corresponding $N$-particle phase space, defined in Eq.~\eqref{eq:m-phase-space}.

We are particularly interested in the cumulative distribution for events that satisfy the selection and for which the observable takes a value smaller than $v$:
\begin{equation}
	\Sigma_{\mathcal{H}}(v) = \sum_{N=n-n_i}^\infty \int d\Phi_N \frac{d\sigma_N}{d\Phi_N}\, \Theta(v - V(q_1,\dots,q_N)) \, \mathcal{H}(q_1,\dots,q_N) \,.
\end{equation}

Following the approach of \textsc{CAESAR}~\cite{Banfi:2004yd}, the cumulant can be rewritten in a factorised form as
\begin{equation}
\Sigma_{\mathcal{H}}(v) \simeq \frac{1}{\sigma_{\mathcal{H}}} \int d\Phi_N \frac{d\sigma_N}{d\Phi_N}\,
\Sigma\big(\{p_{n_i+1},\dots,p_n\},v\big) \,
\mathcal{H}(p_{n_i+1},\dots,p_n) \,,
\end{equation}
where $d\sigma_N/d\Phi_N$ now corresponds to the leading-order cross section for the Born configuration, and the function $\Sigma(\{p\},v)$ encodes the dependence on the observable. \ABquestion{Unsure in the change of momenta from $q\to p$? I have been reading from the original CAESAR paper.}

In this context, $\Sigma(\{p\},v)$ can be interpreted as the probability that a given Born configuration $\{p_{n_i+1},\dots,p_n\}$ produces a value of the observable smaller than $v$. As shown in Sec.~\ref{sec:thrust-example} for thrust, large logarithms appear in the cumulant in the region $v \ll 1$, and so $\Sigma(\{p\},v)$ needs to be resummed to all orders in order to provide reliable results.

While we understand that IRC safe event shapes undergo resummation, there are a few additional properties that observables must have in order to be resummed by the \textsc{ARES} method. Specifically, the observable $V(p_1, p_2, \dots)$ must satisfy the following conditions: 
\begin{enumerate}
\item \textbf{Vanish smoothly}: when an additional emission of momentum $k$ is soft and collinear to a hard leg $\ell$. In this limit, the observable can be written as
\begin{equation}
	V(\{\tilde{p}\},k)\simeq V_{\text{sc}}(\{\tilde{p}\},k)\equiv d_\ell\left(\frac{k_t^{(\ell)}}{Q}\right)^a e^{-b_\ell\eta^{(\ell)}} g_\ell(\phi^{(\ell)})\,,

\end{equation}
where $V_{\text{sc}}$ denotes the observable evaluated in the soft and collinear limit. The set $\{\tilde{p}\}$ represents the Born momenta after recoil from the emission $k$, whose transverse momentum, rapidity, and azimuthal angle with respect to the emitting leg $p_\ell$ are given by $k_t^{(\ell)}$, $\eta^{(\ell)}$, and $\phi^{(\ell)}$, respectively.  The energy $Q$ is an arbitrary hard scale of the process and serves as a reference normalisation for the transverse momenta. \\ \\

Writing observables in this form allows these parameters to be extracted for each leg, enabling the construction of the resummation formula. Further details will be discussed later. \item \textbf{Recursively IRC (rIRC) safe}: this condition requires that, for an ensemble of soft and collinear emissions, the addition of a further emission that is relatively much softer and/or collinear should not significantly modify the value of the observable.  Consider momenta $\kappa_i(\lambda_i)$ such that the observable satisfies
\begin{equation}
\label{eq:rIRC1}
	V(\{\tilde{p}\},\kappa_i(\lambda_i))=\lambda_i\,,
\end{equation}
where $\lambda_i\to0$ in the soft and/or collinear limits, while the azimuthal angles $\phi_i$ of the emissions $\kappa_i(\lambda_i)$ are fixed. The functions $\kappa_i(\lambda_i)$ may differ for each emission but must satisfy Eq.~\eqref{eq:rIRC1}. The conditions for rIRC safety are then:
\begin{enumerate}
	\item The following limit must be well-defined and non-zero:
	\begin{equation}
	\lim_{\varepsilon\to0}\frac{1}{\varepsilon}V(\{\tilde{p}\},\kappa_1(\varepsilon\lambda_1),\dots,\kappa_m(\varepsilon\lambda_m))\,,
	\end{equation}
	which can be interpreted as requiring the soft and collinear scaling of the observable to be independent of the number of emissions.
	\item The following equality must hold:
	\begin{equation}
	\begin{aligned}
	\lim_{\lambda_{m+1}\to0}\lim_{\varepsilon\to0}\frac{1}{\varepsilon}V(&\{\tilde{p}\},\kappa_1(\varepsilon\lambda_1),\dots,\kappa_m(\varepsilon\lambda_m), \kappa_{m+1}(\varepsilon\lambda_{m+1}))\\
		&= \lim_{\varepsilon\to0}\frac{1}{\varepsilon}V(\{\tilde{p}\},\kappa_1(\varepsilon\lambda_1),\dots,\kappa_m(\varepsilon\lambda_m))\,,
	\end{aligned}
	\end{equation}
	showing that the addition of an additional much softer and/or collinear emission must not affect the value of the observable.
\end{enumerate}

\section{Kinematics}
To illustrate the semi-numerical formalism \textsc{ARES}, we focus on a specific example. Since the research in Chapter~5 largely follows the methods of Ref.~\cite{Arpino:2019ozn}, we present the formalism in the context of three-jet event shapes in $e^+e^-$ collisions, in the near-to-planar limit~\cite{Arpino:2019ozn}.

We consider a generic, continuously global, rIRC-safe observable 
$V(\{\tilde{p}\},k_1,\dots,k_n)$, which vanishes in the three-jet limit:
\begin{equation}
V(\{\tilde{p}\}) = 0\,.
\end{equation} 
Here, $\{\tilde{p}\} = \{\tilde{p}_1,\tilde{p}_2,\tilde{p}_3\}$ are the Born-level momenta of the quark, anti-quark, and gluon after recoil from $n$ soft or collinear emissions $k_1,\dots,k_n$. In this sense, $\{\tilde{p}\}$ represent the recoiled final-state momenta.

At Born level, a three-jet event consists of a quark, an anti-quark, and a gluon, with momenta parameterised as
\begin{equation}
\label{eq:3-born-kin}
p_1 = E_1(1,0,0,1)\,,\quad
p_2 = E_2(1,0,\sin\theta_{12},\cos\theta_{12})\,,\quad
p_3 = E_3(1,0,-\sin\theta_{13},\cos\theta_{13})\,,
\end{equation}
where $\theta_{ij}$ is the angle between $p_i$ and $p_j$.

The additional emissions $k$ can be conveniently described using Sudakov variables, such that it can be  decomposed along a pair of light-like momenta $(p_i,p_j)$, which constitute the $(ij)$ dipole, as follows:
\begin{equation}
k^{(ij)}=z^{(i)}p_i+z^{(j)}p_j+\kappa^{(ij)}\cos\phi^{(ij)}n_{\mathrm{in}}^{(ij)}+\kappa^{(ij)}\sin\phi^{(ij)}n_{\mathrm{out}}^{(ij)}\,.
\label{eq:Sudakov-decomp}
\end{equation}
where $z^{(i)}$ and $z^{(j)}$ are energy fractions, and the invariant transverse momentum $\kappa^{(ij)}$ is
\begin{equation}
\left(\kappa^{(ij)}\right)^2=\frac{(2p_i\cdot k)(2p_j\cdot k)}{(2p_i\cdot p_j)}\,.
\end{equation}

The unit vectors $n_{\mathrm{in}}^{(ij)}$ and $n_{\mathrm{out}}^{(ij)}$ define the directions in and orthogonal to the dipole plane:
\begin{equation}
n_{\mathrm{in}}^{(ij)}=\left(\cot\frac{\theta_{ij}}{2},\frac{\vec{n}_i+\vec{n}_j}{\sin\theta_{ij}}\right)\,,\quad n_{\mathrm{out}}^{(ij)}=\left(0,\frac{\vec{n}_i\times\vec{n}_j}{\sin\theta_{ij}}\right)\,,
\end{equation}
where $\vec{n}_\ell\equiv\vec{p}_\ell/E_\ell$ for $\ell=i,j$. 

Finally, the rapidity of the emission relative to the dipole is defined as
\begin{equation}
	\eta^{(ij)}\equiv\frac{1}{2}\ln\frac{z^{(i)}}{z^{(j)}}\,.
\end{equation}
For an emission $k$ collinear to $p_i$ or $p_j$, one can relate the dipole rapidity $\eta^{(ij)}$ to the rapidity $\eta^{(\ell)}$ with respect to the leg $p_\ell$ is
\begin{equation}
	\eta^{(i)}\simeq\eta^{(ij)}+\ln\frac{2E_i}{Q_{ij}}\,,\quad \eta^{(j)}\simeq-\eta^{(ij)}+\ln\frac{2E_j}{Q_{ij}}\,,\quad Q_{ij}^2=2(p_i\cdot p_j)\,.
\end{equation}



\section{NLL Resummation}
In the near-to-planar limit, $v \ll 1$, the cumulant for a three-jet observable can be written in factorised form as
\begin{equation}
\label{eq:3jet-cumu-gen}

\end{equation}
where:
\begin{itemize}
    \item $\mathcal{V}(\{p\})$ encodes the virtual corrections to the Born process, normalised to the Born matrix element squared.
    \item $|\mathcal{M}(\{\tilde{p}\},k_1,\dots,k_n)|^2$ is the squared matrix element for $n$ real emissions off the three-particle Born configuration $\{\tilde{p}\}$.
    \item $S(n)$ is a symmetry factor e.g., $1/n!$ for $n$ identical gluons.
    \item The Lorentz-invariant phase space in $d=4-2\varepsilon$ dimensions is
    \begin{equation}
    [dk] \equiv \frac{d^d k}{(2\pi)^{d-1}} \delta(k^2) \Theta(k^0)\,.
    \end{equation}

 \begin{equation} \alpha_s^{\mathrm{CMW}} = \alpha_s\left(1+\frac{\alpha_s}{2\pi} K\right)+\mathcal{O}(\alpha_s^3)\,,
\end{equation} 
where
\begin{equation}

\end{equation}

 \subsection{The NLL Radiator}
 The Sudakov exponent $R_{\mathrm{NLL}}(v)$ in Eq.~\eqref{eq:NLL-cumulant} is referred to as the NLL radiator. This encodes the probability of no emissions with $V(\{\tilde{p}\},k_i)>v$ and is formed by the combination of virtual corrections with \emph{unresolved} emissions i.e. $V(\{\tilde{p}\},k_i)<\epsilon v$, where $\epsilon$ is a cut off. The radiator is made up of contributions arising from the soft \emph{and} collinear region in the form of double logarithms, as well as single logarithms coming from virtual corrections from soft large-angle and hard-collinear regions. Note that the derivative $R'_{\mathrm{NLL}}$ only includes logarithmic contributions up to NLL accuracy. For a three-jet configuration, the NLL radiator is a sum over the dipoles $(q\bar q)$, $(qg)$ and $(g\bar q)$,
\begin{equation}
\label{eq:NLL-radiator}
\begin{split}
	R_{\mathrm{NLL}}(v)=\sum_{(ij)\in\{(q\bar{q}), (qg), (g\bar{q})\}}C_{(ij)}\left(\sum_{\ell=i,j}\left[r_\ell(L)+r'_\ell(L)\left(\langle \ln(d_\ell g_\ell)\rangle - b_\ell\ln\frac{2E_\ell}{Q}\right) \right. \right. & \\
	\left. \left. +B_\ell T\left(\frac{L}{a+b_\ell}\right)\right] +2\ln\frac{Q_{ij}}{Q}T\left(\frac{L}{a}\right)\right)\,,
\end{split}
\end{equation}
where $Q_{ij}^2=2p_i\!\cdot p_j$ and $C_{(ij)}$ is the colour factor of the dipole,
\begin{equation}
C_{(ij)}=
\begin{cases}
2C_F-C_A\,, & q\bar{q}\,,\\
C_A\,,& qg,\,g\bar{q}\,.
\end{cases}
\end{equation}
The functions $r_\ell(L), r'_\ell(L)$ and $T(L)$ are integrals over the real emission matrix elements and explicit expressions can be found in~\cite{Arpino:2019ozn}. The azimuthal average is
\begin{equation}
\langle f \rangle \equiv \int_0^{2\pi}\frac{d\phi}{2\pi}f\left(\phi\right)\,.
\end{equation}

 \subsection{The NLL Function}
To define the NLL-function, $\mathcal{F}_{\mathrm{NLL}}$ in Eq.~\eqref{eq:NLL-cumulant}, each emission $k_i$ is parametrised by $\{\ell_i,\phi_i^{(\ell_i)},\zeta_i\}$, 
where; $\ell_i$ denotes the leg to which the emission is collinear, $\phi_i^{(\ell_i)}$ is the azimuthal angle with respect to that leg, and $\zeta_i \equiv \frac{V_{\mathrm{sc}}(k_i)}{v}$. For event shapes, the dependence on emission rapidity can be integrated analytically, which in turn can yield a Jacobian factor $R'_{\mathrm{NLL},\ell} \equiv 2 C_\ell\, r'_\ell(L)$, with $C_\ell=C_F$ for quark legs and $C_\ell=C_A$ for gluon legs. With this parametrisation,
  \begin{equation}
    \label{eq:FNLL-formula}
    \mathcal{F}_{\mathrm{NLL}}\left(R'_{\mathrm{NLL}}(v)\right)= \int\! \dZ \,\Theta\left(1-\frac{\Vsc{\{k_i\}}}{v}\right)\,,
  \end{equation}
where the soft-collinear approximation to the observable is defined as
\begin{equation}
  \label{eq:Vsc}
  \frac{\Vsc{k_1,\dots,k_n}}{v}\equiv \lim_{v\to 0}\frac{V({\{\tilde{p}\},k_1,\dots,k_n})}{v}\,,
\end{equation}
which corresponds to approximation of the observable when all emissions $k_1,\dots,k_n$ are simultaneously soft and collinear. Recursive IRC safety guarantees that this limit exists.
Moreover, the soft-collinear measure $\dZ$ is defined as
  \begin{multline}
    \label{eq:dZ-def}
    \int\! \dZ \,G(\{k_i\}) \\ \equiv \lim_{\epsilon\to 0} \epsilon^{\RpNLL} \sum_{n=0}^\infty\frac{1}{n!} \int \prod_{i=1}^n \left(\sum_{\ell_i} R'_{\text{NLL},\ell_i}\int_{\epsilon}^\infty \frac{d\zeta_i}{\zeta_i}\int_0^{2\pi}\frac{d\phi_i^{(\ell_i)}}{2\pi}\right)G(k_1,\dots,k_n)\,.
  \end{multline}
As before, the term with the Sudakov exponent, $\epsilon^{R'_{\mathrm{NLL}}}$, arises from \emph{unresolved} emissions with $\zeta_i<\epsilon$ together with virtual corrections. The remaining terms occur due to rIRC safety, where emissions contributing to 
$V\left(\{\tilde{p}\},k_1,\dots,k_n\right)$ are restricted to the region where $\epsilon v\lesssim
 V(\{\tilde{p}\},k_i)\sim v$ with $\epsilon \ll v$. Together, with the corresponding virtual corrections, they build up $ \mathcal{F}_{\mathrm{NLL}}$.

The $\mathcal{R}_{\ell,0}^{(ij)}(v)$ is calculated to be
\begin{equation}
	\mathcal{R}_{\ell,0}^{(ij)}(v)=R_{\ell,0}^{(ij)}(v)+\left(R_{\ell,0}^{(ij)}\right)'(v)\Big\langle\ln\left(d_\ell^{(ij)}g_\ell(\phi)\right)\Big\rangle+\frac{1}{2}\left(R_{\ell,0}^{(ij)}\right)''(v)\Big\langle\ln^2\left(d_\ell^{(ij)}g_\ell(\phi)\right)\Big\rangle+\dots\,,
\end{equation}
where 
\begin{equation}
	R_{\ell,0}^{(ij)}(v)=-\frac{\lambda}{\alpha_s(Q)\beta_0}g_1^{(\ell)}(\lambda)-g_2^{(\ell)}(\lambda)-\mathfrak{g}_2^{(\ell)}(\lambda,Q_{ij})-\frac{\alpha_s(Q)}{\pi}\left(g_3^{(\ell)}+\mathfrak{g}_3^{(\ell)}(\lambda,Q_{ij})\right)\,,
\end{equation}
and 
\begin{equation}
	d_{\ell}^{(ij)}=d_\ell\left(\frac{Q_{ij}}{Q}\right)^a\left(\frac{Q_{ij}}{2E_\ell}\right)^{b_\ell}\,.
\end{equation}
The second term in Eq.~\eqref{eq:RS-ell} is defined to be
\begin{equation}
	\delta R_{\ell}^{(ij)}(v)=-\frac{\alpha_s(Q)}{\pi}\delta g_3^{(\ell)}(\lambda)\,.
\end{equation}
 The functions $g_1^{(\ell)}(\lambda), g_2^{(\ell)}(\lambda), \mathfrak{g}_2^{(\ell)}(\lambda,Q_{ij}),g_3^{(\ell)}(\lambda) \mathfrak{g}_3^{(\ell)}(\lambda,Q_{ij}),$ and $\delta g_3^{(\ell)}(\lambda)$ are defined in Ref.~\cite{Arpino:2019ozn}.
The hard-collinear radiator is defined as
\begin{equation}
	R_{\mathrm{hc}}(v)=\sum_{\ell}R_{\mathrm{hc},\ell}(v)\,,
\end{equation}
where 
\begin{equation}
R_{\mathrm{hc},\ell}(v)=-h_2^{(\ell)}(\lambda)-\frac{\alpha_s}{\pi}h_3^{(\ell)}(\lambda)\,.
\end{equation}
The functions $h_2^{(\ell)}(\lambda),h_3^{(\ell)}(\lambda)$ can be found in Ref.~\cite{Arpino:2019ozn}.


\subsection{Soft Cumulative Distribution}
The master formula for the soft cumulant at NNLL accuracy is the following: 
\begin{equation}
	\Sigma_{\mathrm{soft}}^{\mathrm{NNLL}}(v)=e^{-R_{\mathrm{NNLL}}(v)}H\left(\{p\},\alpha_s(Q)\right)\left(\mathcal{F}_{\mathrm{NLL}}(\lambda)+\frac{\alpha_s}{\pi}\left(\delta\mathcal{F}_{\mathrm{sc}}+\delta\mathcal{F}_{\mathrm{wa}}+\delta\mathcal{F}_{\mathrm{correl}}+\Delta\mathcal{F}_{\mathrm{wa}}\right)\right)\,,
\end{equation}
where $H\left(\{p\},\alpha_s(Q)\right)$ is the finite hard function which captures all finite terms in the one-loop corrections to the Born event: \todo{change this here}
\begin{equation}
	H\left(\{p\},\alpha_s(Q)\right)=1+\frac{\alpha_s(Q)}{2\pi}H^{(1)}+\mathcal{O}(\alpha_s^2)\,,
\end{equation}
where $H^{(1)}$ is the finite function of order $\alpha_s$ which appears is Eq.~\eqref{eq:Sigma-NNLL-fs}.
The origins of the remaining $\mathcal{F}$-functions are found in the above section. We will now outline how these functions are calculated.

The correction $\delta \mathcal{F}_{\mathrm{sc}}$ requires the approximation given in Eq.~\ref{eq:Vsc}. Additionally, on top of the soft-collinear 
emissions $k_1,\dots,k_n$ that enter any resummation, this function, for which all gluons are widely separated in angle, involve 
an additional soft-collinear emission $k$, parametrised in terms of the previously discussed variables 
$\{\zeta, \ell, \phi^{(\ell)}\}$. The general expression for $\delta \mathcal{F}_{\mathrm{sc}}$ is
\begin{equation}
	\begin{aligned}
		&\delta\mathcal{F}_{\mathrm{sc}}\;=\frac{\pi}{\alpha_s(Q)}\int_0^\infty\frac{d\zeta}{\zeta}\int_0^{2\pi}\frac{d\phi}{2\pi}\sum_{\ell=1}^3\left[\Delta R'_{\ell,\mathrm{NNLL}}+ R''_{\ell,\mathrm{NNLL}}\left(\ln\frac{d_\ell g_\ell(\phi)}{\zeta}-b_\ell\ln\frac{2E_\ell}{Q}\right)\right]\times\\
		&\times\int \dZ  \left[\Theta\left(1-\frac{\Vsc{k,\{k_i\}}}{v}\right)
       -\Theta\left(1-\frac{\Vsc{\{k_i\}}}{v}\right)\Theta(1-\zeta)
     \right] \,,
	\end{aligned}
\end{equation}
 where explicit expressions for $\Delta R'_{\ell,\mathrm{NNLL}}$ and $R''_{\ell,\mathrm{NNLL}}$ can be found in Ref.~\cite{Arpino:2019ozn}. 
 
 The correlated contribution, $\delta \mathcal{F}_{\mathrm{correl}}$, also uses the limit in Eq.~\eqref{eq:Vsc}, however uses it where one 
soft-collinear gluon splits into a $q\bar{q}$ pair or a gluon pair. This additional emission splits into $k_a$ and $k_b$, as shown in Fig.~\ref{fig:splitting}, with further dimensionless variables parametrising the splitting, as shown in Eq.~\eqref{eq:dFcorrel-general}.

 The general expression for $\delta \mathcal{F}_{\mathrm{correl}}$ is 
\begin{equation}
\label{eq:dFcorrel-general}
	\begin{aligned}
	&\delta\mathcal{F}_{\mathrm{correl}}(v)=\\
	&=\sum_{\ell}\frac{\lambda R''_{\ell,\mathrm{NNLL}}}{2a\beta_0\alpha_s(Q)}\int_0^\infty\frac{d\zeta}{\zeta}\int_0^{2\pi}\frac{d\phi^{(\ell)}}{2\pi}\int_0^\infty\frac{d\mu^2}{\mu^2(1+\mu^2)}\int_0^1dz\int_0^{2\pi}\frac{d\phi}{2\pi}\frac{1}{2!}\mathcal{A}^2(z,\mu,\phi)\times\\
	&\int\dZ  \left[\Theta\left(1-\frac{\Vsc{k_a, k_b,\{k_i\}}}{v}\right)
       -\lim_{\mu^2\to0}\Theta\left(1-\frac{\Vsc{k_a+k_b,\{k_i\}}}{v}\right)
     \right] \,,
	\end{aligned}
\end{equation}
where $k_a$ and $k_b$ are soft and collinear to the same leg $\ell$. Here, $\mu$ denotes the rescaled invariant mass of the correlated pair, defined via $m^2\equiv(k_a+k_b)^2$ as
\[
\mu^2=\frac{(k_a+k_b)^2}{(\vec{k}_a^{(ij)}+\vec{k}_b^{(ij)})^2}\,,
\]
and $\mathcal{A}^2$ is the double-emission function,
\[\mathcal{A}^2\equiv C_A(2\mathcal{S}+\mathcal{H}_g)+n_f\mathcal{H}_q\,.
\] 
Explicit definitions of $\vec{k}_a^{(ij)}$, $\vec{k}_b^{(ij)}$, $\mathcal{S}$, 
$\mathcal{H}_g$, and $\mathcal{H}_q$ can be found in Appendix~A 
of Ref.~\cite{Arpino:2019ozn}. 

The soft wide-angled emissions, $\delta\mathcal{F}_{\mathrm{wa}}$ and $\Delta\mathcal{F}_{\mathrm{wa}}$ rely on the assumption from Eq.~\ref{eq:Vsc}, while $\delta\mathcal{F}_{\mathrm{wa}}$ also depends on  $V_{\mathrm{wa}}$, i.e. the observable in the limit when a single soft gluon $k$ is emitted at large angles. Both functions make use of Sudakov variables, $k^{(ij)}$ as defined in Eq.~\eqref{eq:Sudakov-decomp}.  The general expression for $\delta\mathcal{F}_{\mathrm{wa}}$ is given by
\begin{equation}
\begin{aligned}
	&\delta\mathcal{F}_{\mathrm{wa}}(\lambda)=\sum_{(ij)}C_{(ij)}\frac{\alpha_s(v^{1/a}\,Q)}{a\,\alpha_s(Q)}\int_0^\infty\frac{d\zeta}{\zeta}\int_{-\infty}^\infty d\eta^{(ij)}\int_0^{2\pi}\frac{d\phi^{(ij)}}{2\pi}\times\\
	&\times \int \dZ  \left[\Theta\left(1-\frac{\Vwa{k^{(ij)},\{k_i\}}}{v}\right)
       -\Theta\left(1-\frac{\Vsc{k^{(ij)},\{k_i\}}}{v}\right)\,
     \right] \,.
\end{aligned}
\end{equation}
The general expression for $\Delta\mathcal{F}_{\mathrm{wa}}$ is given by
\begin{equation}
\begin{aligned}
	&\Delta\mathcal{F}_{\mathrm{wa}}(\lambda)=\sum_{(ij)}C_{(ij)}\frac{\alpha_s(v^{1/a}Q)}{a\,\alpha_s(Q)}\ln\frac{Q_{ij}}{Q}\int_0^\infty\frac{d\zeta}{\zeta}\int_{-\infty}^\infty d\eta^{(ij)}\int_0^{2\pi}\frac{d\phi^{(ij)}}{2\pi}\times\\
	&\times \int \dZ  \left[\Theta\left(1-\lim_{v\to0}\frac{\Vsc{k,\{k_i\}}}{v}\right)
       -\Theta\left(1-\lim_{v\to0}\frac{\Vsc{\{k_i\}}}{v}\right)\Theta(1-\zeta)
     \right] \,.
\end{aligned}
\end{equation}

\subsection{Hard-collinear Cumulative Distribution}
At NNLL accuracy, the hard-collinear cumulant is
\begin{equation}
	\Sigma_{\mathrm{hc}}^{\mathrm{NNLL}}(v)=e^{-R_{\mathrm{NNL}}(v)}H\left(\{p\},\alpha_s(Q)\right)\left(\mathcal{F}_{\mathrm{NLL}}(\lambda)\sum_{\ell=1}^3 C_{\mathrm{hc},\ell}^{(1)}+\frac{\alpha_s}{\pi}\left(\delta\mathcal{F}_{\mathrm{rec}}+\delta\mathcal{F}_{\mathrm{hc}}+\mathcal{T}\left(\{p\}\right)\Delta\mathcal{F}_{\mathrm{rec}}\right)\right)\,,
\end{equation}
where $\mathcal{T}(\{p\})$ is the pre-factor found from the spin correlations and the description ofthe  $\mathcal{F}$-functions are found in the above section.
We will now define these functions explicitly. 

The general expression for $\delta \mathcal{F}_{\mathrm{hc}}$ is 
 \begin{multline}
   \label{eq:dFhc-general}
   \delta \mathcal{F}_{\mathrm{hc}}(\lambda) \equiv \sum_\ell \frac{\alpha_s\left(Qv^{1/(a+b_\ell)}\right)}{\alpha_s(Q)(a+b_\ell)}\int_0^\infty\frac{d\zeta}{\zeta}\int_0^{\pi}\frac{d\phi^{(\ell)}}{\pi}\int_0^1dz^{(\ell)} \langle P_{f_\ell}\left(z^{(\ell)},0\right)\rangle\times\\
    \times \int\! \dZ \, \times \left[\Theta\left(1-\frac{\Vsc{k,\{k_i\}}}{v}\right)
       -\Theta\left(1-\frac{\Vsc{\{k_i\}}}{v}\right)\Theta(1-\zeta)
     \right] \,,
 \end{multline}
where $\langle P_{f_\ell}\left(z^{(\ell)},\epsilon\right)\rangle$ is the 
azimuthally averaged splitting function, which depends on $f_\ell$, the 
flavour of leg $\ell$:
\begin{equation}
\langle P_{f_\ell}(z^{(\ell)},\epsilon)\rangle \equiv 
\begin{cases}
C_F\left(-2+(1-\epsilon)\right)\,, & f_\ell = q, \bar{q} \\[6pt]
C_A\left(z(1-z)-2\right)+T_R n_f\left[1-\dfrac{2z(1-z)}{1-\epsilon}\right]\,, & f_\ell = g\,.
\end{cases}
\end{equation}

The recoil corrections, $\delta\mathcal{F}_{\mathrm{rec}}, \Delta\mathcal{F}_{\mathrm{rec}}$ rely on the approximation found in Eq.~\ref{eq:Vsc} along with the following
\begin{equation}
  \label{eq:Vhc}
  \frac{\Vhc{k_{\mathrm{hc}}, k_1,\dots,k_n}}{v}\equiv \lim_{v\to 0}\frac{V({\{\tilde{p}\},k_{\mathrm{hc}},k_1,\dots,k_n})}{v}\,,
\end{equation}
where $k_{\mathrm{hc}}$ denotes the hard-collinear gluon. 

The general expression for $\delta\mathcal{F}_{\mathrm{rec}}$ is
\begin{equation}
\begin{aligned}
	&\delta\mathcal{F}_{\mathrm{rec}}(\lambda)=\sum_{\ell=1}^3\frac{\alpha_s(Qv^{1/(a+b_\ell)})}{\alpha_s(Q)(a+b_\ell)}\int_0^\infty\frac{d\zeta}{\zeta}\int_0^{\pi}\frac{d\phi^{(\ell)}}{2\pi}\int_0^1dz^{(\ell)}\left(\frac{2C_\ell}{z^{(\ell)}}+\langle P_{f_\ell}(z^{(\ell)},0)\rangle\right)\times\\
	&\times \int \dZ  \left[\Theta\left(1-\frac{\Vhc{k,\{k_i\}}}{v}\right)
       -\Theta\left(1-\frac{\Vsc{k,\{k_i\}}}{v}\right)
     \right] \,,
\end{aligned}
\end{equation}
where $C_{q,\bar{q}}=C_F$ and $C_g=C_A$. \\ \\
 The general expression for $\Delta\mathcal{F}_{\mathrm{rec}}$ is
\begin{equation}
\begin{aligned}
	&\Delta\mathcal{F}_{\mathrm{rec}}(\lambda)=\frac{\alpha_s(Qv^{1/(a+b_g)})}{\alpha_s(Q)(a+b_g)}\int_0^\infty\frac{d\zeta}{\zeta}\int_0^{\pi}\frac{d\phi^{(\ell)}}{2\pi}\int_0^1dz^{(\ell)} \Delta P_g(z^{(\ell)},\phi^{(\ell)})\times\\
	&\times \int \dZ  \left[\Theta\left(1-\frac{\Vhc{k,\{k_i\}}}{v}\right)
       -\Theta(1-\zeta)\Theta\left(1-\frac{\Vsc{\{k_i\}}}{v}\right)
     \right] \,,
\end{aligned}
\end{equation}
where the un-averaged splitting function for the gluon 
\begin{equation}
	 \Delta P_g(z,\phi;\epsilon)=4z(1-z)(2(1-\epsilon)\cos^2\phi-1)\left(\frac{C_A}{2}-\frac{T_Rn_f}{1-\epsilon}\right)\,.
\end{equation}







\subsection{Additive Observables}
Here we will briefly discuss a certain class of observables which take an additive form. Additivity is the concept that an observable can be written as a the sum of contributions of individual emissions i.e. for soft and collinear emissions $k_1,\dots , k_n$, this means 
\begin{equation}
\Vsc{k_1,\dots , k_n}=\sum_{i=1}^n\Vsc{k_i}\,.
\end{equation}
Additive observables allow $\FNLL$ and for all terms in Eq.~\eqref{eq:dFNNLL} to be calculated analytically. The equations which govern these observables can be found in Ref.~\cite{Arpino:2019ozn}. Examples of additive observables are: thrust introduced in Sec.~\ref{sec:thrust-example} and the one-jettiness and transverse thrust introduced in Ch~\ref{ch:hhZjet}.
